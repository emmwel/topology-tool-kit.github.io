361  const SimplexId *const offsets,
362  const bool splitTree) const {
363  if(triplets.empty())
364  return;
365 
366  const auto lt = [=](const SimplexId a, const SimplexId b) -> bool {
367  return offsets[a] < offsets[b];
368  };
369 
370  // Sorting step
371  const auto cmp = [=](const triplet &t1, const triplet &t2) {
372  const SimplexId s1 = std::get<0>(t1);
373  const SimplexId s2 = std::get<0>(t2);
374  const SimplexId m1 = std::get<2>(t1);
375  const SimplexId m2 = std::get<2>(t2);
376  if(s1 != s2)
377  return lt(s1, s2) != splitTree;
378  else // s1 == s2

380  };

394  return;

396 
397  const auto lt = [=](const SimplexId a, const SimplexId b) -> bool {
398  return offsets[a] < offsets[b];
399  };
400 
401  const auto getRep = [&](SimplexId v) -> SimplexId {
402  auto r = vertexRepresentatives[v][0];
403  while(r != v) {
404  v = r;
405  r = vertexRepresentatives[v][0];
406  }
407  return r;
408  };
409 
410  for(const auto &t : triplets) {
411  SimplexId r1 = getRep(std::get<1>(t));
412  SimplexId r2 = getRep(std::get<2>(t));
413  if(r1 != r2) {
414  SimplexId s = std::get<0>(t);
415  numberOfPairs++;
416 
417  // Add pair
418  if(splitTree) {
419  // r1 = min(r1, r2), r2 = max(r1, r2)
420  if(lt(r2, r1)) {
421  std::swap(r1, r2);
422  }
423  // pair saddle-max: s -> min(r1, r2);
424  pairs.emplace_back(s, r1, 2);
425 
426  } else {

428  if(lt(r1, r2)) {
429  std::swap(r1, r2);
430  }
431  // pair min-saddle: max(r1, r2) -> s;
432  pairs.emplace_back(r1, s, 0);
433  }
434 
435  vertexRepresentatives[std::get<1>(t)][0] = r2;

437  }
438  }

699 ttk::SimplexId ttk::ProgressiveTopology::propageFromSaddles(
700  const SimplexId vertexId,
701  std::vector<Lock> &vertLock,
702  std::vector<polarity> &toPropage,
703  std::vector<std::vector<SimplexId>> &vertexRepresentatives,
704  std::vector<std::vector<SimplexId>> &saddleCC,
705  std::vector<polarity> &isUpdated,
706  std::vector<SimplexId> &globalExtremum,
707  const SimplexId *const offsets,
708  const bool splitTree) const {
709 
710  auto &toProp = toPropage[vertexId];
711  auto &reps = vertexRepresentatives[vertexId];
712  auto &updated = isUpdated[vertexId];
713 
714  const auto gt = [=](const SimplexId a, const SimplexId b) {
715  return (offsets[a] > offsets[b]) == splitTree;
716  };
717 
718  if(updated) {
719  return reps[0];
720  }
721 
722  if(this->threadNumber_ > 1) {
723  vertLock[vertexId].lock();
724  }
725 
726  if(toProp) { // SADDLE POINT

728  reps.clear();
729  reps.reserve(CC.size());
730  for(size_t r = 0; r < CC.size(); r++) {
731  SimplexId neighborId = -1;
732  SimplexId localId = CC[r];
733  multiresTriangulation_.getVertexNeighbor(vertexId, localId, neighborId);
734  SimplexId ret = propageFromSaddles(
735  neighborId, vertLock, toPropage, vertexRepresentatives, saddleCC,
736  isUpdated, globalExtremum, offsets, splitTree);
737  reps.emplace_back(ret);
738  }
739 
740  if(reps.size() > 1) {
741  // sort & remove duplicate elements
742  std::sort(reps.begin(), reps.end(), gt);
743  const auto last = std::unique(reps.begin(), reps.end());
744  reps.erase(last, reps.end());
745  }
746 
747  updated = 255;
748  if(this->threadNumber_ > 1) {
749  vertLock[vertexId].unlock();
750  }
751 
752  return reps[0];
753 
754  } else {
755 
756  SimplexId ret = vertexId;
757  SimplexId neighborNumber
758  = multiresTriangulation_.getVertexNeighborNumber(vertexId);
759  SimplexId maxNeighbor = vertexId;
760  for(SimplexId i = 0; i < neighborNumber; i++) {
761  SimplexId neighborId = -1;
762  multiresTriangulation_.getVertexNeighbor(vertexId, i, neighborId);
763  if(gt(neighborId, maxNeighbor)) {
764  maxNeighbor = neighborId;
765  }
766  }
767  if(maxNeighbor != vertexId) { // not an extremum
768  ret = propageFromSaddles(maxNeighbor, vertLock, toPropage,
769  vertexRepresentatives, saddleCC, isUpdated,
770  globalExtremum, offsets, splitTree);
771  } else {
772 #ifdef TTK_ENABLE_OPENMP
773  const auto tid = omp_get_thread_num();
774 #else
775  const auto tid = 0;
776 #endif // TTK_ENABLE_OPENMP
777  if(gt(vertexId, globalExtremum[tid])) {
778  globalExtremum[tid] = vertexId;
779  }
780  }
781  reps.resize(1);
782  reps[0] = ret;
783  updated = 255;
784  if(this->threadNumber_ > 1) {
785  vertLock[vertexId].unlock();
786  }
787  return ret;
788  }

1043 void ttk::ProgressiveTopology::updateDynamicLink(
1044  DynamicTree &link,
1045  std::vector<std::pair<polarity, polarity>> &vlp,
1046  std::vector<std::pair<SimplexId, SimplexId>> &vl) const {
1047 
1048  std::vector<SimplexId> monotony_changes_list{};
1049 
1050  for(size_t neighborId = 0; neighborId < vlp.size(); neighborId++) {
1051  const polarity isBroken = vlp[neighborId].second;
1052  if(isBroken != 0) {
1053  monotony_changes_list.emplace_back(neighborId);
1054  }
1055  }
1056 
1057  // loop on the link
1058  // for each edge that shares n0
1059  // if only one break and different polarity : remove
1060  // else if only one break and same polarity : insert
1061  // else : do nothing
1062  std::vector<std::vector<std::pair<SimplexId, SimplexId>>> edgesToInsertLater(
1063  vlp.size());
1064  std::vector<std::vector<std::pair<SimplexId, SimplexId>>> edgesToRemoveLater(
1065  vlp.size());
1066 
1067  for(size_t e = 0; e < vl.size(); e++) {
1068  const SimplexId n0 = vl[e].first;
1069  const SimplexId n1 = vl[e].second;
1070  const polarity isBroken0 = vlp[n0].second;
1071  const polarity isBroken1 = vlp[n1].second;
1072 
1073  if(isBroken0 != 0 and isBroken1 == 0) {
1074  if(vlp[n0].first != vlp[n1].first) {
1075  edgesToInsertLater[n0].emplace_back(n0, n1);
1076  } else {
1077  edgesToRemoveLater[n0].emplace_back(n0, n1);
1078  }
1079  } else if(isBroken0 == 0 and isBroken1 != 0) {
1080  if(vlp[n0].first != vlp[n1].first) {
1081  edgesToInsertLater[n1].emplace_back(n1, n0);
1082  } else {
1083  edgesToRemoveLater[n1].emplace_back(n1, n0);
1084  }
1085  }
1086  }
1087 
1088  // REMOVE EDGES:
1089  for(const auto brokenNode : monotony_changes_list) {
1090  vlp[brokenNode].first = ~vlp[brokenNode].first;
1091  vlp[brokenNode].second = 0;
1092 
1093  link.getNode(brokenNode)->evert();
1094  for(const auto &edge : edgesToRemoveLater[brokenNode]) {
1095  link.removeEdge(edge.first, edge.second);
1096  }
1097  }
1098  if(!edgesToRemoveLater.empty()) {
1099  // reconnect link
1100  for(const auto &edge : vl) {
1101  if(vlp[edge.first].first == vlp[edge.second].first) {
1102  link.insertEdge(edge.first, edge.second);
1103  }
1104  }
1105  }
1106 
1107  // INSERT EDGES
1108  for(const auto brokenNode : monotony_changes_list) {
1109  for(const auto &edge : edgesToInsertLater[brokenNode]) {
1110  link.insertEdge(edge.first, edge.second);
1111  }
1112  }
1113 }
