228 // Can't be parallel on a vector of bool !!

241  template <typename triangulationType>
242  void
243  Mesh<triangulationType>::preSortTriangles(const VertCompFN &lowerThan) {
244 // require edges to be already sorted
245 //
246 // Not sure we can parallelie on a vector of bitfields
247 #ifdef TTK_ENABLE_OPENMP
248 #pragma omp parallel for schedule(static)
249 #endif
250  for(idCell ti = 0; ti < nbTriangles_; ++ti) {
251  idEdge e0, e1, e2;
252  getTriangleEdge(ti, 0, e0);
253  getTriangleEdge(ti, 1, e1);
254  getTriangleEdge(ti, 2, e2);
255 
256  if(compareEdges(e0, e1, lowerThan)) {
257  // 1 2 3
258  // 1 3 2
259  // 2 3 1
260  if(compareEdges(e1, e2, lowerThan)) {
261  // 0 : 1 2 3
262  trianglesSortId_[ti] = 0;
263  } else if(compareEdges(e0, e2, lowerThan)) {
264  // 1 : 1 3 2
265  trianglesSortId_[ti] = 1;
266  } else {
267  // 3 : 2 3 1
268  trianglesSortId_[ti] = 3;
269  }
270  } else {
271  // 2 1 3
272  // 3 1 2
273  // 3 2 1
274  if(compareEdges(e0, e2, lowerThan)) {
275  // 2 : 2 1 3
276  trianglesSortId_[ti] = 2;
277  } else if(compareEdges(e1, e2, lowerThan)) {
278  // 4 : 3 1 2
279  trianglesSortId_[ti] = 4;
280  } else {
281  // 5 : 3 2 1
282  trianglesSortId_[ti] = 5;
283  }
284  }
285  }
