316  std::vector<std::vector<size_t>> idx(nInputs);
317 
318  for(size_t i = 0; i < nInputs; i++) {
319  double local_min_persistence = std::numeric_limits<double>::min();

321  for(int j = 0; j < bidder_diags[i].size(); j++) {

323  double persistence = b.getPersistence();
324  if(persistence >= 0.0 && persistence <= prev_min_persistence) {
325  candidates_to_be_added[i].emplace_back(j);
326  idx[i].emplace_back(idx[i].size());
327  persistences.emplace_back(persistence);
328  }
329  }

331  return ((persistences[a] > persistences[b])
332  || ((persistences[a] == persistences[b]) && (a > b)));
333  };
334  std::sort(idx[i].begin(), idx[i].end(), cmp);
335  const auto size = candidates_to_be_added[i].size();
336  if(size >= max_points_to_add) {
337  double last_persistence_added
338  = persistences[idx[i][max_points_to_add - 1]];
339  if(last_persistence_added > local_min_persistence) {
340  local_min_persistence = last_persistence_added;
341  }
342  }
343  if(i == 0) {
344  new_min_persistence = local_min_persistence;
345  } else {
346  if(local_min_persistence < new_min_persistence) {
347  new_min_persistence = local_min_persistence;
348  }
349  }

355  const double persistence = b.getPersistence();
