79  int interpolateContinuousScalarField(
80  const T *data,
81  T *output,
82  const triangulationType &inputTriangl,
83  const ExplicitTriangulation &outputTriangl) const {
84  static_assert(
85  std::is_floating_point<T>::value, "Floating point type required.");
86  const auto nOutVerts = this->getNumberOfVertices();
87  if(nOutVerts < 0 || nOutVerts != nVertices_ + nEdges_ + nTriangles_) {
88  return 1;
89  }
90 
91  // copy data on parent vertices
92  std::copy(data, data + nVertices_, output);
93 
94  // interpolate on edges
95  for(SimplexId i = 0; i < nEdges_; ++i) {
96  SimplexId a{}, b{};
97  inputTriangl.getEdgeVertex(i, 0, a);
98  inputTriangl.getEdgeVertex(i, 0, b);
99  output[nVertices_ + i] = (data[a] + data[b]) / T{2.0};
100  }
101 
102  // interpolate on triangle barycenters
103  for(SimplexId i = 0; i < nTriangles_; ++i) {
104  SimplexId a{}, b{}, c{};
105  inputTriangl.getTriangleVertex(i, 0, a);
106  inputTriangl.getTriangleVertex(i, 1, b);
107  inputTriangl.getTriangleVertex(i, 2, c);
108  output[nVertices_ + nEdges_ + i]
109  = (data[a] + data[b] + data[c]) / T{3.0};
110  }
111  return 0;
112  }

192 template <typename triangulationType>
193 int ttk::BarycentricSubdivision::subdiviseTriangulation(
194  const triangulationType &inputTriangl) {
195 
196  // not implemented for dimension >= 3
197  if(inputTriangl.getDimensionality() >= 3) {
198  this->printErr("Not yet implemented for dimension 3 and above");
199  return 1;
200  }
201 
202  const SimplexId newPoints{nVertices_ + nEdges_ + nTriangles_};
203  const size_t dataPerPoint{3};
204  points_.clear();
205  points_.resize(newPoints * dataPerPoint);
206 
207  const size_t dataPerCell{3};
208  const size_t newTrianglesPerParent{6};
209  cells_connectivity_.clear();
210  cells_offsets_.clear();
211  cells_connectivity_.resize(nTriangles_ * newTrianglesPerParent * dataPerCell);
212  cells_offsets_.resize(nTriangles_ * newTrianglesPerParent + 1);
213 
214  pointId_.clear();
215  pointId_.resize(newPoints);
216 
217  pointDim_.clear();
218  pointDim_.resize(newPoints);
219 
220  // set input point coordinates and ids
221  for(SimplexId i = 0; i < nVertices_; ++i) {
222  inputTriangl.getVertexPoint(
223  i, points_[i + 0], points_[i + 1], points_[i + 2]);
224  pointId_[i] = i;
225  }
226 
227  // reserve memory for new points
228  points_.reserve(newPoints * dataPerPoint);
229 
230  // generate edge middles
231  for(SimplexId i = 0; i < nEdges_; ++i) {
232  // edge vertices
233  SimplexId a{}, b{};
234  inputTriangl.getEdgeVertex(i, 0, a);
235  inputTriangl.getEdgeVertex(i, 1, b);
236 
237  std::array<float, 3> pa{}, pb{}, mid{};
238  inputTriangl.getVertexPoint(a, pa[0], pa[1], pa[2]);
239  inputTriangl.getVertexPoint(b, pb[0], pb[1], pb[2]);
240 
241  mid[0] = (pa[0] + pb[0]) / 2.0F;
242  mid[1] = (pa[1] + pb[1]) / 2.0F;
243  mid[2] = (pa[2] + pb[2]) / 2.0F;
244 
245  const size_t offset = dataPerPoint * (nVertices_ + i);
246  points_[offset + 0] = mid[0];
247  points_[offset + 1] = mid[1];
248  points_[offset + 2] = mid[2];
249  pointId_[nVertices_ + i] = i;
250  pointDim_[nVertices_ + i] = 1;
251  }
252 
253  // generate triangle barycenters
254  for(SimplexId i = 0; i < nTriangles_; ++i) {
255  // triangle vertices
256  SimplexId a{}, b{}, c{};
257  inputTriangl.getTriangleVertex(i, 0, a);
258  inputTriangl.getTriangleVertex(i, 1, b);
259  inputTriangl.getTriangleVertex(i, 2, c);
260 
261  std::array<float, 3> pa{}, pb{}, pc{}, bary{};
262  inputTriangl.getVertexPoint(a, pa[0], pa[1], pa[2]);
263  inputTriangl.getVertexPoint(b, pb[0], pb[1], pb[2]);
264  inputTriangl.getVertexPoint(c, pc[0], pc[1], pc[2]);
265 
266  bary[0] = (pa[0] + pb[0] + pc[0]) / 3.0F;
267  bary[1] = (pa[1] + pb[1] + pc[1]) / 3.0F;
268  bary[2] = (pa[2] + pb[2] + pc[2]) / 3.0F;
269 
270  const size_t offset = dataPerPoint * (nVertices_ + nEdges_ + i);
271  points_[offset + 0] = bary[0];
272  points_[offset + 1] = bary[1];
273  points_[offset + 2] = bary[2];
274  pointId_[nVertices_ + nEdges_ + i] = i;
275  pointDim_[nVertices_ + nEdges_ + i] = 2;
276  }
277 
278  LongSimplexId off_id = 0, co_id = 0;
279  // subdivise every triangle
280  for(SimplexId i = 0; i < nTriangles_; ++i) {
281  // id of triangle barycenter
282  SimplexId bary = nVertices_ + nEdges_ + i;
283 
284  for(SimplexId j = 0; j < inputTriangl.getTriangleEdgeNumber(i); ++j) {
285  // edge id
286  SimplexId e{};
287  inputTriangl.getTriangleEdge(i, j, e);
288 
289  // id of middle of edge e
290  SimplexId em = nVertices_ + e;
291 
292  // edge vertices
293  SimplexId a{}, b{};
294  inputTriangl.getEdgeVertex(e, 0, a);
295  inputTriangl.getEdgeVertex(e, 1, b);
296 
297  // Ad triangle a - em - bary
298  cells_offsets_[off_id] = co_id;
299  off_id++;
300  cells_connectivity_[co_id + 0] = a;
301  cells_connectivity_[co_id + 1] = em;
302  cells_connectivity_[co_id + 2] = bary;
303  co_id += 3;
304 
305  // Ad triangle b - em - bary
306  cells_offsets_[off_id] = co_id;
307  off_id++;
308  cells_connectivity_[co_id + 0] = b;
309  cells_connectivity_[co_id + 1] = em;
310  cells_connectivity_[co_id + 2] = bary;
311  co_id += 3;
312  }
313  }
314  // Last offset
315  cells_offsets_[off_id] = co_id;
316 
317  return 0;
318 }
