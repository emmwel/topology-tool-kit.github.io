337 bool RangeDrivenOctree::segmentIntersection(
338  const std::pair<double, double> &p0,
339  const std::pair<double, double> &p1,
340  const std::pair<double, double> &q0,
341  const std::pair<double, double> &q1) const {
342 
343  // NOTE: [q0, q1] is axis aligned
344  bool horizontal = true;
345  double x = 0, y = 0;
346 
347  if(q0.first == q1.first)
348  horizontal = false;
349 
350  double denP = p1.first - p0.first;
351  if(!denP)
352  denP = DBL_EPSILON;
353 
354  double P = (p1.second - p0.second) / denP;
355  if(!P)
356  P = DBL_EPSILON;
357  double bP = p1.second - P * p1.first;
358 
359  if(horizontal) {
360  y = q0.second;
361  x = (y - bP) / P;
362  } else {
363  x = q0.first;
364  y = P * x + bP;
365  }
366 
367  // does the intersection lands on the range segment?
368  if(horizontal) {
369  if((x < fmin(q0.first, q1.first)) || (x > fmax(q0.first, q1.first))) {
370  return false;
371  }
372  } else {
373  if((y < fmin(q0.second, q1.second)) || (y > fmax(q0.second, q1.second))) {
374  return false;
375  }
376  }
377 
378  // does it lands on the polygon segment's projection?
379  if((x < fmin(p0.first, p1.first) || (x > fmax(p0.first, p1.first)))
380  && ((y < fmin(p0.second, p1.second)
381  || (y > fmax(p0.second, p1.second))))) {
382  return false;
383  }
384 
385  return true;
