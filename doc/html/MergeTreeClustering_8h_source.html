605  void computeCentroids(std::vector<ftm::FTMTree_MT *> &trees,
606  std::vector<ftm::MergeTree<dataType>> &centroids,
607  matchingVector &outputMatching,
608  std::vector<double> &alphas,
609  std::vector<int> &clusteringAssignment,
610  std::vector<ftm::FTMTree_MT *> &trees2,
611  std::vector<ftm::MergeTree<dataType>> &centroids2,
612  matchingVector &outputMatching2) {
613  Timer t_clust;
614 
615  // Run
616  int noCentroidsT = centroids.size();
617  bool converged = false;
618  dataType inertia = -1;
619  dataType minInertia = std::numeric_limits<dataType>::max();
620  int cptBlocked = 0;
621  noIterationC_ = 0;
622  std::vector<std::tuple<int, int>> assignmentC;
623  std::vector<dataType> bestDistanceT(
624  trees.size(), std::numeric_limits<dataType>::max());
625  while(not converged) {
626  ++noIterationC_;
627 
628  printMsg(debug::Separator::L1);
629  std::stringstream ssIter;
630  ssIter << "Itration " << noIterationC_;
631  printMsg(ssIter.str());
632 
633  // --- Assignment
634  Timer t_assignment;
635  assignmentCentroids<dataType>(
636  trees, centroids, assignmentC, bestDistanceT, trees2, centroids2);
637  auto t_assignment_time = t_assignment.getElapsedTime();
638  printMsg("Assignment", 1, t_assignment_time, this->threadNumber_);
639 
640  // --- Update
641  Timer t_update;
642  bool trees1Updated = true, trees2Updated = true;
643  trees1Updated
644  = updateCentroids<dataType>(trees, centroids, alphas, assignmentC);
645  if(trees2.size() != 0)
646  trees2Updated = updateCentroids<dataType>(
647  trees2, centroids2, alphas, assignmentC);
648  auto t_update_time = t_update.getElapsedTime();
649  printMsg("Update", 1, t_update_time, this->threadNumber_);
650 
651  // --- Check convergence
652  dataType currentInertia = 0;
653  for(auto distance : bestDistanceT)
654  currentInertia += distance * distance;
655  converged = std::abs((double)(inertia - currentInertia)) < 0.01;
656  inertia = currentInertia;
657  std::stringstream ss3;
658  ss3 << "Inertia : " << inertia;
659  printMsg(ss3.str());
660 
661  minInertia = std::min(minInertia, inertia);
662  if(not converged) {
663  cptBlocked += (minInertia < inertia) ? 1 : 0;
664  converged = (cptBlocked >= 10);
665  }
666 
667  // Converged if barycenters were not updated (same assignment than last
668  // iteration)
669  converged = converged or (not trees1Updated and not trees2Updated);
670 
671  // --- Reset vectors
672  if(not converged) {
673  assignmentC.clear();
674  bestDistanceT = std::vector<dataType>(
675  trees.size(), std::numeric_limits<dataType>::max());
676  }
677  }
678 
679  // Final processing
680  printMsg(debug::Separator::L1);
681  printMsg("Final assignment");
682  matchingVector matchingsC(noCentroidsT);
683  matchingVector matchingsC2(noCentroidsT);
684  finalAssignmentCentroids<dataType>(trees, centroids, matchingsC,
685  assignmentC, bestDistanceT, trees2,
686  centroids2, matchingsC2);
687  for(auto dist : bestDistanceT)
688  finalDistances_.push_back(dist);
689  dataType currentInertia = 0;
690  for(auto distance : bestDistanceT)
691  currentInertia += distance * distance;
692  std::stringstream ss;
693  ss << "Inertia : " << currentInertia;
694  printMsg(ss.str());
695 
696  // Manage output
697  std::vector<int> cptCentroid(centroids.size(), 0);
698  for(auto asgn : assignmentC) {
699  int centroid = std::get<0>(asgn);
700  int tree = std::get<1>(asgn);
701  // std::cout << centroid << " " << tree << std::endl;
702  clusteringAssignment[tree] = centroid;
703  outputMatching[centroid][tree]
704  = matchingsC[centroid][cptCentroid[centroid]];
705  if(trees2.size() != 0)
706  outputMatching2[centroid][tree]
707  = matchingsC2[centroid][cptCentroid[centroid]];
708  ++cptCentroid[centroid];
709  }
710 
711  auto clusteringTime = t_clust.getElapsedTime() - addDeletedNodesTime_;
712  printMsg("Total", 1, clusteringTime, this->threadNumber_);
713  }
