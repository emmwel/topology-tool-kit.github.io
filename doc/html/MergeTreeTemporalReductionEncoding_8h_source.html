51  template <class dataType>
52  dataType computeL2Distance(std::vector<dataType> &img1,
53  std::vector<dataType> &img2,
54  bool emptyFieldDistance = false) {
55  size_t noPoints = img1.size();
56 
57  std::vector<dataType> secondField = img2;
58  if(emptyFieldDistance)
59  secondField = std::vector<dataType>(noPoints, 0);
60 
61  dataType distance = 0;
62 
63  for(size_t i = 0; i < noPoints; ++i)
64  distance += std::pow((img1[i] - secondField[i]), 2);
65 
66  distance = std::sqrt(distance);
67 
68  return distance;

177  toRemoved = std::min(toRemoved, (int)(mTrees.size() - 3));
178

181  for(size_t j = 0; j < fieldL2_[i].size(); ++j)
182  images[i].push_back(static_cast<dataType>(fieldL2_[i][j]));
183 
184  for(int iter = 0; iter < toRemoved; ++iter) {
185  dataType bestCost = std::numeric_limits<dataType>::max();
186  int bestMiddleIndex = -1;
187  ftm::MergeTree<dataType> bestBarycenter;
188  std::vector<std::tuple<ftm::MergeTree<dataType>, int>>
189  bestBarycentersOnPath;
190  std::vector<dataType> bestBarycenterL2;
191  std::vector<std::tuple<std::vector<dataType>, int>>

195  printMsg("Compute barycenter for each pair of trees",
196  debug::Priority::VERBOSE);
197  unsigned int index1 = 0, index2 = 0;
198  while(index2 != mTrees.size() - 1) {
199 
200  // Get index in the middle
201  int middleIndex = index1 + 1;
202  while(treeRemoved[middleIndex])
203  ++middleIndex;
204 
205  // Get second index
206  index2 = middleIndex + 1;
207  while(treeRemoved[index2])
208  ++index2;
209 
210  // Compute barycenter
211  printMsg("Compute barycenter", debug::Priority::VERBOSE);
212  double alpha = computeAlpha(index1, middleIndex, index2);
213  ftm::MergeTree<dataType> barycenter;
214  std::vector<dataType> barycenterL2;
215  if(not useL2Distance_)
216  barycenter = computeBarycenter<dataType>(
217  mTrees[index1], mTrees[index2], alpha);
218  else
219  barycenterL2 = computeL2Barycenter<dataType>(
220  images[index1], images[index2], alpha);
221 
222  // - Compute cost
223  // Compute distance with middleIndex
224  printMsg(
225  "Compute distance with middleIndex", debug::Priority::VERBOSE);
226  dataType cost;
227  if(not useL2Distance_)
228  cost = computeDistance<dataType>(barycenter, mTrees[middleIndex]);
229  else
230  cost
231  = computeL2Distance<dataType>(barycenterL2, images[middleIndex]);
232 
233  // Compute distances of previously removed trees on the path
234  printMsg("Compute distances of previously removed trees",
235  debug::Priority::VERBOSE);
236  std::vector<std::tuple<ftm::MergeTree<dataType>, int>>
237  barycentersOnPath;
238  std::vector<std::tuple<std::vector<dataType>, int>>
239  barycentersL2OnPath;
240  for(unsigned int i = 0; i < 2; ++i) {
241  int toReach = (i == 0 ? index1 : index2);
242  int offset = (i == 0 ? -1 : 1);
243  int tIndex = middleIndex + offset;
244  while(tIndex != toReach) {
245 
246  // Compute barycenter
247  double alphaT = computeAlpha(index1, tIndex, index2);
248  ftm::MergeTree<dataType> barycenterP;
249  std::vector<dataType> barycenterPL2;
250  if(not useL2Distance_)
251  barycenterP = computeBarycenter<dataType>(
252  mTrees[index1], mTrees[index2], alphaT);
253  else
254  barycenterPL2 = computeL2Barycenter<dataType>(
255  images[index1], images[index2], alphaT);
256 
257  // Compute distance
258  dataType costP;
259  if(not useL2Distance_)
260  costP = computeDistance<dataType>(barycenterP, mTrees[tIndex]);
261  else
262  costP
263  = computeL2Distance<dataType>(barycenterPL2, images[tIndex]);
264 
265  // Save results
266  if(not useL2Distance_)
267  barycentersOnPath.push_back(
268  std::make_tuple(barycenterP, tIndex));
269  else
270  barycentersL2OnPath.push_back(
271  std::make_tuple(barycenterPL2, tIndex));
272  cost += costP;
273  tIndex += offset;
274  }
275  }
276 
277  if(cost < bestCost) {
278  bestCost = cost;
279  bestMiddleIndex = middleIndex;
280  if(not useL2Distance_) {
281  bestBarycenter = barycenter;
282  bestBarycentersOnPath = barycentersOnPath;
283  } else {
284  bestBarycenterL2 = barycenterL2;
285  bestBarycentersL2OnPath = barycentersL2OnPath;
286  }
287  }
288 
289  // Go to the next index
290  index1 = middleIndex;
291  }
292 
293  // Removed the tree with the lowest cost
294  printMsg(
295  "Removed the tree with the lowest cost", debug::Priority::VERBOSE);
296  removed.push_back(bestMiddleIndex);
297  treeRemoved[bestMiddleIndex] = true;
298  if(not useL2Distance_) {
299  barycenters[bestMiddleIndex] = bestBarycenter;
300  for(auto &tup : bestBarycentersOnPath)
301  barycenters[std::get<1>(tup)] = std::get<0>(tup);
302  } else {
303  barycentersL2[bestMiddleIndex] = bestBarycenterL2;
304  for(auto &tup : bestBarycentersL2OnPath)
305  barycentersL2[std::get<1>(tup)] = std::get<0>(tup);
306  }
307  }
308  }

311  std::vector<int> execute(std::vector<ftm::MergeTree<dataType>> &mTrees,
312  std::vector<double> &emptyTreeDistances,
313  std::vector<ftm::MergeTree<dataType>> &allMT) {

327 
328  // --- Execute

331  std::vector<int> removed;
332  if(not useCustomTimeVariable_) {
333  timeVariable_.clear();
334  for(size_t i = 0; i < mTrees.size(); ++i)
335  timeVariable_.push_back(i);
336  }
337  temporalSubsampling<dataType>(
338  mTrees, removed, barycenters, barycentersL2);
339 
340  // --- Concatenate all trees/L2Images
341  std::vector<std::vector<dataType>> images(fieldL2_.size());
342  for(size_t i = 0; i < fieldL2_.size(); ++i)
343  for(size_t j = 0; j < fieldL2_[i].size(); ++j)
344  images[i].push_back(static_cast<dataType>(fieldL2_[i][j]));
345 
346  for(auto &mt : mTrees)
347  allMT.push_back(mt);
348  std::vector<bool> removedB(mTrees.size(), false);
349  for(auto r : removed)
350  removedB[r] = true;
351  for(unsigned int i = 0; i < barycenters.size(); ++i)
352  if(removedB[i]) {
353  if(not useL2Distance_)
354  allMT.push_back(barycenters[i]);
355  else
356  images.push_back(barycentersL2[i]);
357  }
358 
359  // --- Compute empty tree distances
360  unsigned int distMatSize
361  = (not useL2Distance_ ? allMT.size() : images.size());
362  for(unsigned int i = 0; i < distMatSize; ++i) {
363  dataType distance;
364  if(not useL2Distance_)
365  distance = computeDistance<dataType>(allMT[i], allMT[i], true);
366  else
367  distance = computeL2Distance<dataType>(images[i], images[i], true);
368  emptyTreeDistances.push_back(distance);
369  }
370 
371  // --- Postprocessing
372  if(not useL2Distance_) {
373  for(unsigned int i = 0; i < allMT.size(); ++i)
374  postprocessingPipeline<dataType>(&(allMT[i].tree));
375  for(unsigned int i = 0; i < mTrees.size(); ++i)
376  postprocessingPipeline<dataType>(&(mTrees[i].tree));
377  }
378 
379  // --- Print results
380  std::stringstream ss, ss2, ss3;
381  ss << "input size = " << mTrees.size();
382  printMsg(ss.str());
383  ss2 << "output size = "
384  << mTrees.size() - (distMatSize - mTrees.size());
385  printMsg(ss2.str());
386  ss3 << "removed : ";
387  for(unsigned int i = 0; i < removed.size(); ++i) {
388  auto r = removed[i];
389  ss3 << r;
390  if(i < removed.size() - 1)
391  ss3 << ", ";
392  }
393  printMsg(ss3.str());
394 
395  sort(removed.begin(), removed.end());
396 
397  printMsg("Encoding", 1, t_tempSub.getElapsedTime(), this->threadNumber_);
398 
399  return removed;
400  }
