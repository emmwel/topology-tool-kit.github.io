160 void ttk::ftm::FTMTreePP::computePairs(
161  ftm::FTMTree_MT *tree,
162  std::vector<std::tuple<SimplexId, SimplexId, scalarType>> &pairs) {
163  auto getParentNode = [&](const idNode current) {
164  const idSuperArc parentArc = tree->getNode(current)->getUpSuperArcId(0);
165  return tree->getSuperArc(parentArc)->getUpNodeId();
166  };
167 
168  ::std::queue<idNode> toSee;
169 
170  // start at the leaves
171  auto &vectLeaves = (not useCustomTree) ? tree->getLeaves() : customTreeLeaves;
172  for(auto nid : vectLeaves) {
173  toSee.emplace(nid);
174  }
175 
176  while(!toSee.empty()) {
177  const idNode current = toSee.front();
178  toSee.pop();

181  createPairs<scalarType>(current, pairs, tree, ftm::nullVertex);
182  clearPendingNodes(current);
183  continue;
184  } else {
185  clearPendingNodes(current);
186  }
