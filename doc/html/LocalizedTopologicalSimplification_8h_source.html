774 
775  localVertexSequence[q++] = v;
776 
777  IT nNeighbors = triangulation->getVertexNeighborNumber(v);
778  for(IT n = 0; n < nNeighbors; n++) {
779  IT u;
780  triangulation->getVertexNeighbor(v, n, u);
781 
782  // if u is in segment and has not already been added to the queue
783  if(segmentation[u] == segmentId && localOrder[u] < 0) {
784  queue.emplace(localOrder[u], u);
785  localOrder[u] = 0;
786  }
787  }
788  }
789 
790  if(performSuperlevelSetPropagation) {
791  // skip first idx as it correponds to saddle
792  for(IT i = 1; i <= nSegmentVertices; i++)
793  localOrder[localVertexSequence[i]] = -i;
794  } else {
795  IT order = -nSegmentVertices;
796  // skip last idx as it correponds to saddle
797  for(IT i = 0; i < nSegmentVertices; i++)
798  localOrder[localVertexSequence[i]] = order++;
799  }
800 
801  return 1;
802  }

804  template <typename IT, class TT>
805  int computeLocalOrderOfSegment(IT *localOrder,
806 
807  const Propagation<IT> *propagation,
808  const TT *triangulation,
809  const IT *segmentation,
810  const IT *inputOrder) const {
811  // quick espace for small segments
812  if(propagation->segmentSize == 1) {
813  localOrder[propagation->segment[0]] = -2;
814  return 1;
815  }
816 
817  // init local order by input order
818  // Note: force orders to be smaller than 0 for in situ updates in
819  // propagation procedures
820  IT nVertices = triangulation->getNumberOfVertices();
821  for(const auto &v : propagation->segment)
822  localOrder[v] = inputOrder[v] - nVertices;
823 
824  const IT &extremumIndex = propagation->criticalPoints[0];
825  const IT &saddleIndex = propagation->criticalPoints.back();
826 
827  // vector to record boundary
828  std::vector<IT> boundary;
829 
830  // make enough room for segment + saddle
831  std::vector<IT> localVertexSequence(propagation->segmentSize + 1);
832 
833  int status = 1;
834  bool containsResidualExtrema = true;
835  bool performSuperlevelSetPropagation = true;
836  while(containsResidualExtrema) {
837  propagation->nIterations++;
838 
839  if(this->debugLevel_ > 3 && propagation->nIterations == 20)
840  this->printWrn("Unable to converge with less than 20 iterations!");
841 
842  // execute superlevel set propagation
843  status = this->computeLocalOrderOfSegmentIteration<IT, TT>(
844  localOrder, localVertexSequence.data(),
845 
846  performSuperlevelSetPropagation, triangulation, segmentation,
847  extremumIndex, boundary, propagation->segment, saddleIndex);
848  if(!status)
849  return 0;
850 
851  performSuperlevelSetPropagation = !performSuperlevelSetPropagation;
852 
853  IT boundaryWriteIdx = 0;
854  IT nResidualMaxima = 0;
855  IT nResidualMinima = 0;
856 
857  for(const auto &v : propagation->segment) {
858  bool isOnSegmentBoundary = false;
859  bool hasSmallerNeighbor = false;
860  bool hasLargerNeighbor = false;
861 
862  const auto &vOrder = localOrder[v];
863 
864  IT nNeighbors = triangulation->getVertexNeighborNumber(v);
865  for(IT n = 0; n < nNeighbors; n++) {
866  IT u;
867  triangulation->getVertexNeighbor(v, n, u);
868 
869  if(u == saddleIndex) {
870  hasLargerNeighbor = true;
871  continue;
872  }
873 
874  // if u is not inside segment -> v is on segment boundary
875  if(segmentation[u] != extremumIndex) {
876  isOnSegmentBoundary = true;
877  } else {
878  if(localOrder[u] > vOrder)
879  hasLargerNeighbor = true;
880  else
881  hasSmallerNeighbor = true;
882  }
883  }
884 
885  if(!hasLargerNeighbor)
886  nResidualMaxima++;
887 
888  if(isOnSegmentBoundary) {
889  localVertexSequence[boundaryWriteIdx++] = v;
890  } else if(!hasSmallerNeighbor) {
891  nResidualMinima++;
892  }
893  }
894 
895  containsResidualExtrema = nResidualMinima > 0 || nResidualMaxima > 0;
896 
897  if(containsResidualExtrema && boundary.size() == 0) {
898  boundary.resize(boundaryWriteIdx);
899  for(IT i = 0; i < boundaryWriteIdx; i++) {
900  boundary[i] = localVertexSequence[i];
901  }
902  }
903  }
904 
905  return 1;
906  }
