168  const double scalarMax

170  const double delta = (scalarMax - scalarMin) / (params_.samplingLvl + 1);
171

181

184  pointIds[0] = arcData.points[downVertId];
185  } else {
186  triangulation_->getVertexPoint(
187  downVertId, pointCoord[0], pointCoord[1], pointCoord[2]);
188  pointIds[0] = points->InsertNextPoint(pointCoord);
189  arcData.points.emplace(downVertId, pointIds[0]);
190  arcData.setPointInfo(graph, arcId, pointIds[0]);
191  }
192 
193  float sum[3]{};
194  int chunk = 0;
195  double scalarLimit = scalarMin + delta;
196 
197  for(const idVertex regV : arc.segmentation()) {
198 
199  // for regular mask to be set by the boundary vertices correctly
200  if(regV == downVertId || regV == upVertId)
201  continue;
202

204  regV, pointCoord[0], pointCoord[1], pointCoord[2]);
205  const double scalar = inputScalars_->GetTuple1(regV);
206  if(scalar < scalarLimit) {
207  sum[0] += pointCoord[0];
208  sum[1] += pointCoord[1];
209  sum[2] += pointCoord[2];
210  ++chunk;
211  } else {
212  if(chunk) {
213  sum[0] /= chunk;
214  sum[1] /= chunk;
215  sum[2] /= chunk;
216 
217  // do not use memorized points as even if a point already exisit it is
218  // from another vertex and should not be used
219  pointIds[1] = points->InsertNextPoint(sum);
220  arcData.points.emplace(regV, pointIds[1]);
221  arcData.setPointInfo(graph, arcId, pointIds[1], true);
222

224  vtkIdType nextCellId

226  arcData.setArcInfo(graph, arcId, nextCellId);
227  }
228 
229  pointIds[0] = pointIds[1];
230  }
231 
232  // reset for next iteration
233  scalarLimit += delta;
234  sum[0] = 0;
235  sum[1] = 0;
236  sum[2] = 0;
237  chunk = 0;
238  }
239  }

242  pointIds[1] = arcData.points[upVertId];
243  } else {
244  triangulation_->getVertexPoint(
245  upVertId, pointCoord[0], pointCoord[1], pointCoord[2]);

371 int ttkFTRGraph::getSkeletonArcs(const ttk::ftr::Graph &graph,
372  vtkUnstructuredGrid *outputSkeletonArcs) {
373  const idSuperArc nbArcs = graph.getNumberOfArcs();
374 
375  idSuperArc nbFinArc = 0;
376  switch(params_.samplingLvl) {
377  case -1:
378  // loops may create more arcs
379  nbFinArc = triangulation_->getNumberOfVertices() * 1.5;
380  break;
381  case 0:
382  nbFinArc = graph.getNumberOfVisibleArcs();
383  break;
384  default:
385  nbFinArc = graph.getNumberOfVisibleArcs() * (params_.samplingLvl + 1);
386  break;
387  }
388 
389  ttk::ftr::ArcData arcData(nbFinArc);
390  vtkNew<vtkUnstructuredGrid> arcs{};
391  vtkNew<vtkPoints> points{};
392 
393  for(idSuperArc arcId = 0; arcId < nbArcs; ++arcId) {
394  if(!graph.getArc(arcId).isVisible())
395  continue;
396 
397  switch(params_.samplingLvl) {
398  case -1:
399  addCompleteSkeletonArc(graph, arcId, points, arcs, arcData);
400  break;
401  case 0:
402  addDirectSkeletonArc(graph, arcId, points, arcs, arcData);
403  break;
404  default:
405  addSampledSkeletonArc(graph, arcId, points, arcs, arcData);
406  break;
407  }
408  }
409 
410  arcs->SetPoints(points);
411  arcData.addArrays(arcs, params_);
412  outputSkeletonArcs->ShallowCopy(arcs);
413
