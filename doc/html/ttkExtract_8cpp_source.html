325 int computeMask(signed char *mask,
326 
327  const std::vector<double> &pivotValues,
328  const size_t &nValues,
329  const DT *values,
330  const ttkExtract::VALIDATION_MODE &validationMode,
331  const size_t &threadNumber) {
332 
333  const size_t nPivotValues = pivotValues.size();
334  std::vector<DT> pivotValuesMin(nPivotValues);
335  std::vector<DT> pivotValuesMax(nPivotValues);
336 
337  const DT delta = std::numeric_limits<DT>::is_integer
338  ? 1
339  : std::numeric_limits<DT>::epsilon();
340 
341  for(size_t i = 0; i < nPivotValues; i++) {
342 
343  switch(validationMode) {
344  case ttkExtract::VALIDATION_MODE::LESS_THEN: {
345  pivotValuesMin[i] = std::numeric_limits<DT>::lowest();
346  pivotValuesMax[i] = ((DT)pivotValues[i]) - delta;
347  break;
348  }
349 
350  case ttkExtract::VALIDATION_MODE::LESS_EQUAL_THEN: {
351  pivotValuesMin[i] = std::numeric_limits<DT>::lowest();
352  pivotValuesMax[i] = ((DT)pivotValues[i]);
353  break;
354  }
355 
356  case ttkExtract::VALIDATION_MODE::EQUAL:
357  case ttkExtract::VALIDATION_MODE::UNEQUAL: {
358  pivotValuesMin[i] = ((DT)pivotValues[i]);
359  pivotValuesMax[i] = ((DT)pivotValues[i]);
360  break;
361  }
362 
363  case ttkExtract::VALIDATION_MODE::GREATER_EQUAL_THEN: {
364  pivotValuesMin[i] = ((DT)pivotValues[i]);
365  pivotValuesMax[i] = std::numeric_limits<DT>::max();
366  break;
367  }
368 
369  case ttkExtract::VALIDATION_MODE::GREATER_THEN: {
370  pivotValuesMin[i] = ((DT)pivotValues[i]) + delta;
371  pivotValuesMax[i] = std::numeric_limits<DT>::max();
372  break;
373  }
374  }
375  }
376 
377  int status = computeMask_<DT>(
378  mask, nValues, values, pivotValuesMin, pivotValuesMax, threadNumber);
379 
380  if(validationMode == ttkExtract::VALIDATION_MODE::UNEQUAL)
381  for(size_t i = 0; i < nValues; i++)
382  mask[i] = mask[i] == 0 ? 1 : 0;
383 
384  return status;
385 }

685 int ttkExtract::RequestData(vtkInformation *ttkNotUsed(request),
686  vtkInformationVector **inputVector,
687  vtkInformationVector *outputVector) {
688  // Get Input to Output
689  auto input = vtkDataObject::GetData(inputVector[0]);
690  auto output = vtkDataObject::GetData(outputVector);
691 
692  // -------------------------------------------------------------------------
693  // Replace Variables in ExpressionString (e.g. {time[2]})
694  // -------------------------------------------------------------------------
695  std::string finalExpressionString;
696  {
697  std::string errorMsg;
698  if(!ttkUtils::replaceVariables(this->GetExpressionString(),
699  input->GetFieldData(), finalExpressionString,
700  errorMsg)) {
701  this->printErr(errorMsg);
702  return 0;
703  }
704  }
705 
706  std::vector<double> values;
707  ttkUtils::stringListToDoubleVector(finalExpressionString, values);
708 
709  auto mode = this->ExtractionMode;
710  if(mode == EXTRACTION_MODE::AUTO) {
711  if(input->IsA("vtkMultiBlockDataSet"))
712  mode = EXTRACTION_MODE::BLOCKS;
713  else if(input->IsA("vtkTable"))
714  mode = EXTRACTION_MODE::ROWS;
715  else {
716  this->printErr("Unable to automatically determine extraction mode.");
717  return 0;
718  }
719  }
720 
721  // in case of array or geometry extraction iterate over vtkMultiBlockDataSet
722  auto inputAsMB = vtkSmartPointer<vtkMultiBlockDataSet>::New();
723  auto outputAsMB = vtkSmartPointer<vtkMultiBlockDataSet>::New();
724  size_t nBlocks;
725  if(mode != EXTRACTION_MODE::BLOCKS && mode != EXTRACTION_MODE::BLOCK_TUPLES
726  && input->IsA("vtkMultiBlockDataSet")) {
727  inputAsMB->ShallowCopy(input);
728  nBlocks = inputAsMB->GetNumberOfBlocks();
729 
730  for(size_t b = 0; b < nBlocks; b++) {
731  auto inputBlock = inputAsMB->GetBlock(b);
732 
733  vtkSmartPointer<vtkDataObject> outputBlock;
734  if(mode == EXTRACTION_MODE::BLOCKS && !this->MaskOnly)
735  outputBlock = vtkSmartPointer<vtkUnstructuredGrid>::New();
736  else
737  outputBlock
738  = vtkSmartPointer<vtkDataObject>::Take(inputBlock->NewInstance());
739 
740  outputAsMB->SetBlock(b, outputBlock);
741  }
742  output->ShallowCopy(outputAsMB);
743  } else {
744  inputAsMB->SetBlock(0, input);
745  outputAsMB->SetBlock(0, output);
746  nBlocks = 1;
747  }
748 
749  switch(mode) {
750  case EXTRACTION_MODE::BLOCKS: {
751  if(!this->ExtractBlocks(output, input, values, false))
752  return 0;
753  break;
754  }
755  case EXTRACTION_MODE::BLOCK_TUPLES: {
756  if(!this->ExtractBlocks(output, input, values, true))
757  return 0;
758  break;
759  }
760  case EXTRACTION_MODE::ROWS: {
761  if(!this->ExtractRows(output, input, values))
762  return 0;
763  break;
764  }
765  case EXTRACTION_MODE::GEOMETRY: {
766  for(size_t b = 0; b < nBlocks; b++)
767  if(!this->ExtractGeometry(
768  outputAsMB->GetBlock(b), inputAsMB->GetBlock(b), values))
769  return 0;
770  break;
771  }
772  case EXTRACTION_MODE::ARRAY_VALUES: {
773  for(size_t b = 0; b < nBlocks; b++)
774  if(!this->ExtractArrayValues(
775  outputAsMB->GetBlock(b), inputAsMB->GetBlock(b), values))
776  return 0;
777  break;
778  }
779  case EXTRACTION_MODE::ARRAYS: {
780  for(size_t b = 0; b < nBlocks; b++)
781  if(!this->ExtractArray(
782  outputAsMB->GetBlock(b), inputAsMB->GetBlock(b), values))
783  return 0;
784  break;
785  }
786  default: {
787  this->printErr("Unsupported Extraction Mode");
788  return 0;
789  }
790  }
791 
792  this->printMsg(ttk::debug::Separator::L1);
793 
794  return 1;
795 }
