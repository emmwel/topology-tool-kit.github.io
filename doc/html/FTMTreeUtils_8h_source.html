214  if(nodeCorr[nodeOrigin] == -1)
215  treeNew->makeNode(nodeOriginIndex);
216  else

218  int nodeIndex = treeNew->getNumberOfNodes();
219  if(nodeCorr[node] == -1)
220  treeNew->makeNode(nodeIndex);
221  else

226  treeNew->getNode(nodeIndex)->setOrigin(nodeOriginIndex);

230  newScalarsVector[nodeIndex] = tree->getValue<dataType>(node);
231  nodeCorr[nodeOrigin] = nodeOriginIndex;
232  nodeCorr[node] = nodeIndex;
233  } else {

235  treeNew->makeNode(nodeCpt);
236  if(!tree->isLeaf(node)) {
