136  template <typename dataType>
137  typename KDTree<dataType>::KDTreeMap
138  KDTree<dataType>::build(dataType *data,
139  const int &ptNumber,
140  const int &dimension,
141  const int weight_number) {
142  KDTreeMap correspondance_map(ptNumber);
143  // First, perform a argsort on the data
144  // initialize original index locations
145  for(int axis = 0; axis < dimension; axis++) {
146  coords_min_.push_back(std::numeric_limits<dataType>::lowest());
147  coords_max_.push_back(std::numeric_limits<dataType>::max());
148  }
149  std::vector<int> idx(ptNumber);
150  for(int i = 0; i < ptNumber; i++) {
151  idx[i] = i;
152  }
153  // sort indexes based on comparing values in coordinates
154  sort(idx.begin(), idx.end(), [&](int i1, int i2) {
155  return data[dimension * i1 + coords_number_]
156  < data[dimension * i2 + coords_number_];
157  });
158  int median_loc = (int)(ptNumber - 1) / 2;
159  int median_idx = idx[median_loc];
160  correspondance_map[median_idx] = this;
161 
162  for(int axis = 0; axis < dimension; axis++) {
163  coordinates_.push_back(data[dimension * median_idx + axis]);
164  }
165  for(int w = 0; w < weight_number; w++) {
166  weight_.push_back(0);
167  min_subweights_.push_back(0);
168  }
169  id_ = median_idx;
170  parent_ = nullptr;
171  level_ = 0;
172 
173  if(idx.size() > 2) {
174  // Build left leaf
175  std::vector<int> idx_left(median_loc);
176  for(int i = 0; i < median_loc; i++) {
177  idx_left[i] = idx[i];
178  }
179 
180  this->left_ = std::unique_ptr<KDTree>(
181  new KDTree(this, (coords_number_ + 1) % dimension, true));
182  this->left_->buildRecursive(data, idx_left, ptNumber, dimension, this,
183  correspondance_map, weight_number);
184  }
185 
186  if(idx.size() > 1) {
187  // Build right leaf
188  std::vector<int> idx_right(ptNumber - median_loc - 1);
189  for(int i = 0; i < ptNumber - median_loc - 1; i++) {
190  idx_right[i] = idx[i + median_loc + 1];
191  }
192  this->right_ = std::unique_ptr<KDTree>(
193  new KDTree(this, (coords_number_ + 1) % dimension, false));
194  this->right_->buildRecursive(data, idx_right, ptNumber, dimension, this,
195  correspondance_map, weight_number);
196  }
197 
198  return correspondance_map;
199  }

326  template <typename dataType>
327  void KDTree<dataType>::recursiveGetKClosest(
328  const unsigned int k,
329  const std::vector<dataType> &coordinates,
330  KDTreeMap &neighbours,
331  std::vector<dataType> &costs,
332  const int weight_index) {
333  // 1- Look wether or not to include the current point in the nearest
334  // neighbours
335  dataType cost = this->cost(coordinates);
336  cost += weight_[weight_index];
337 
338  if(costs.size() < k) {
339  neighbours.push_back(this);
340  costs.push_back(cost);
341  } else {
342  // 1.1- Find the most costly amongst neighbours
343  std::vector<int> idx(k);
344  for(unsigned int i = 0; i < k; i++) {
345  idx[i] = i;
346  }
347  int idx_max_cost = *std::max_element(
348  idx.begin(), idx.end(),
349  [&costs](int &a, int &b) { return costs[a] < costs[b]; });
350  dataType max_cost = costs[idx_max_cost];
351 
352  // 1.2- If the current KDTree is less costly, put it in the neighbours and
353  // update costs.
354  if(cost < max_cost) {
355  costs[idx_max_cost] = cost;
356  neighbours[idx_max_cost] = this;
357  }
358  }
359 
360  // 2- Recursively visit KDTrees that are worth it
361  if(left_) {
362  dataType max_cost = *std::max_element(costs.begin(), costs.end());
363  dataType &min_subweight = left_->min_subweights_[weight_index];
364  dataType d_min = this->distanceToBox(*left_, coordinates);
365  if(costs.size() < k || d_min + min_subweight < max_cost) {
366  // 2.2- It is possible that there exists a point in this subtree that is
367  // less costly than max_cost
368  left_->recursiveGetKClosest(
369  k, coordinates, neighbours, costs, weight_index);
370  }
371  }
372 
373  if(right_) {
374  dataType max_cost = *std::max_element(costs.begin(), costs.end());
375  dataType &min_subweight = right_->min_subweights_[weight_index];
376  dataType d_min = this->distanceToBox(*right_, coordinates);
377  if(costs.size() < k || d_min + min_subweight < max_cost) {
378  // 2.2- It is possible that there exists a point in this subtree that is
379  // less costly than max_cost
380  right_->recursiveGetKClosest(
381  k, coordinates, neighbours, costs, weight_index);
382  }
383  }
384  return;
385  }
