210  template <typename dataType>
211  void AssignmentAuction<dataType>::runAuctionRound(
212  std::vector<std::vector<dataType>> &cMatrix) {
213  std::queue<int> unassignedBidders{};
214  for(unsigned int i = 0; i < bidderAssignments.size(); ++i)
215  unassignedBidders.push(i);
216 
217  while(!unassignedBidders.empty()) {
218  int bidderId = unassignedBidders.front();
219  unassignedBidders.pop();
220 
221  // Get good with highest value
222  dataType bestValue = std::numeric_limits<dataType>::lowest();
223  dataType bestSecondValue = std::numeric_limits<dataType>::lowest();
224  int bestGoodId = -1;
225  for(unsigned int goodId = 0; goodId < goodPrices.size(); ++goodId) {
226  if(cMatrix[bidderId][goodId] == static_cast<dataType>(-1))
227  continue;
228  dataType goodPrice = goodPrices[goodId];
229  dataType value = -cMatrix[bidderId][goodId] - goodPrice;
230  if(value > bestValue) {
231  bestSecondValue = bestValue;
232  bestValue = value;
233  bestGoodId = goodId;
234  } else if(value > bestSecondValue)
235  bestSecondValue = value;
236  }
237 
238  // Update assignments
239  bidderAssignments[bidderId] = bestGoodId;
240  if(goodAssignments[bestGoodId] != -1)
241  unassignedBidders.push(goodAssignments[bestGoodId]);
242  goodAssignments[bestGoodId] = bidderId;
243 
244  // Update price
245  double delta = abs<dataType>(bestValue - bestSecondValue) + epsilon;
246  goodPrices[bestGoodId] = goodPrices[bestGoodId] + delta;
247 
248  // printVector(goodPrices);
249  }
