459 template <typename triangulationType>
460 int ttk::AbstractMorseSmaleComplex::getDescendingSeparatrices1(
461  const std::vector<Cell> &criticalPoints,
462  std::vector<Separatrix> &separatrices,
463  std::vector<std::vector<Cell>> &separatricesGeometry,
464  const triangulationType &triangulation) const {
465 
466  std::vector<SimplexId> saddleIndexes;
467  const SimplexId numberOfCriticalPoints = criticalPoints.size();
468  for(SimplexId i = 0; i < numberOfCriticalPoints; ++i) {
469  const Cell &criticalPoint = criticalPoints[i];
470 
471  if(criticalPoint.dim_ == 1)
472  saddleIndexes.push_back(i);
473  }
474  const SimplexId numberOfSaddles = saddleIndexes.size();
475 
476  // estimation of the number of separatrices, apriori :
477  // numberOfAscendingPaths=2, numberOfDescendingPaths=2
478  const SimplexId numberOfSeparatrices = 4 * numberOfSaddles;
479  separatrices.resize(numberOfSeparatrices);
480  separatricesGeometry.resize(numberOfSeparatrices);
481 
482  // apriori: by default construction, the separatrices are not valid
483 #ifdef TTK_ENABLE_OPENMP
484 #pragma omp parallel for num_threads(threadNumber_)
485 #endif
486  for(SimplexId i = 0; i < numberOfSaddles; ++i) {
487  const SimplexId saddleIndex = saddleIndexes[i];
488  const Cell &saddle = criticalPoints[saddleIndex];
489 
490  // add descending vpaths
491  {
492  const Cell &saddle1 = saddle;
493 
494  for(int j = 0; j < 2; ++j) {
495  const int shift = j + 2;
496 
497  SimplexId vertexId;
498  triangulation.getEdgeVertex(saddle1.id_, j, vertexId);
499 
500  std::vector<Cell> vpath;
501  vpath.push_back(saddle1);
502  discreteGradient_.getDescendingPath(
503  Cell(0, vertexId), vpath, triangulation);
504 
505  const Cell &lastCell = vpath.back();
506  if(lastCell.dim_ == 0 and discreteGradient_.isCellCritical(lastCell)) {
507  const SimplexId separatrixIndex = 4 * i + shift;
508 
509  separatricesGeometry[separatrixIndex] = std::move(vpath);
510  separatrices[separatrixIndex]
511  = Separatrix(true, saddle, lastCell, false, separatrixIndex);
512  }
513  }
514  }
515  }
516 
517  return 0;
518 }

713 int ttk::AbstractMorseSmaleComplex::setAscendingSegmentation(
714  const std::vector<Cell> &criticalPoints,
715  std::vector<SimplexId> &maxSeeds,
716  SimplexId *const morseSmaleManifold,
717  SimplexId &numberOfMaxima,
718  const triangulationType &triangulation) const {
719 
720  const SimplexId numberOfVertices = triangulation.getNumberOfVertices();
721  std::fill(morseSmaleManifold, morseSmaleManifold + numberOfVertices, -1);
722 
723  const SimplexId numberOfCells = triangulation.getNumberOfCells();
724  std::vector<SimplexId> morseSmaleManifoldOnCells(numberOfCells, -1);
725  const int cellDim = triangulation.getDimensionality();
726 
727  // get the seeds : maxima
728  const SimplexId numberOfCriticalPoints = criticalPoints.size();
729  for(SimplexId i = 0; i < numberOfCriticalPoints; ++i) {
730  const Cell &criticalPoint = criticalPoints[i];
731 
732  if(criticalPoint.dim_ == cellDim)
733  maxSeeds.push_back(criticalPoint.id_);
734  }
735  const SimplexId numberOfSeeds = maxSeeds.size();
736  numberOfMaxima = numberOfSeeds;
737 
738  // Triangulation method pointers for 3D
739  auto getCellFace = &triangulationType::getCellTriangle;
740  auto getFaceStarNumber = &triangulationType::getTriangleStarNumber;
741  auto getFaceStar = &triangulationType::getTriangleStar;
742  if(cellDim == 2) {
743  // Triangulation method pointers for 2D
744  getCellFace = &triangulationType::getCellEdge;
745  getFaceStarNumber = &triangulationType::getEdgeStarNumber;
746  getFaceStar = &triangulationType::getEdgeStar;
747  }
748 
749 #ifdef TTK_ENABLE_OPENMP
750 #pragma omp parallel for num_threads(threadNumber_) schedule(dynamic)
751 #endif
752  for(SimplexId i = 0; i < numberOfSeeds; ++i) {
753  std::queue<SimplexId> bfs;
754 
755  // push the seed
756  {
757  const SimplexId seedId = maxSeeds[i];
758  bfs.push(seedId);
759  }
760 
761  // BFS traversal
762  while(!bfs.empty()) {
763  const SimplexId cofacetId = bfs.front();
764  bfs.pop();
765 
766  if(morseSmaleManifoldOnCells[cofacetId] == -1) {
767  morseSmaleManifoldOnCells[cofacetId] = i;
768 
769  for(int j = 0; j < (cellDim + 1); ++j) {
770  SimplexId facetId = -1;
771  (triangulation.*getCellFace)(cofacetId, j, facetId);
772 
773  SimplexId starNumber = (triangulation.*getFaceStarNumber)(facetId);
774  for(SimplexId k = 0; k < starNumber; ++k) {
775  SimplexId neighborId = -1;
776  (triangulation.*getFaceStar)(facetId, k, neighborId);
777 
778  if(neighborId == cofacetId) {
779  continue;
780  }
781 
782  const SimplexId pairedCellId = discreteGradient_.getPairedCell(
783  Cell(cellDim, neighborId), triangulation, true);
784 
785  if(pairedCellId == facetId)
786  bfs.push(neighborId);
787  }
788  }
789  }
790  }
791  }
792 
793  // put segmentation infos from cells to points
794 #ifdef TTK_ENABLE_OPENMP
795 #pragma omp parallel for num_threads(threadNumber_)
796 #endif
797  for(SimplexId i = 0; i < numberOfVertices; ++i) {
798  SimplexId starId;
799  triangulation.getVertexStar(i, 0, starId);
800  morseSmaleManifold[i] = morseSmaleManifoldOnCells[starId];
801  }
802 
803  return 0;
804 }

877 int ttk::AbstractMorseSmaleComplex::setFinalSegmentation(
878  const SimplexId numberOfMaxima,
879  const SimplexId ttkNotUsed(numberOfMinima),
880  const SimplexId *const ascendingManifold,
881  const SimplexId *const descendingManifold,
882  SimplexId *const morseSmaleManifold,
883  const triangulationType &triangulation) const {
884 
885  const size_t nVerts = triangulation.getNumberOfVertices();
886 
887  // associate a unique "sparse region id" to each (ascending, descending) pair
888 
889 #ifdef TTK_ENABLE_OPENMP

891 #endif // TTK_ENABLE_OPENMP

893  const auto d = ascendingManifold[i];
894  const auto a = descendingManifold[i];
895  if(a == -1 || d == -1) {

897  } else {

899  }

904  morseSmaleManifold, morseSmaleManifold + nVerts);

909  sparseRegionIds.erase(last, sparseRegionIds.end());
910 
911  // "sparse region id" -> "dense region id"
912  std::map<SimplexId, size_t> sparseToDenseRegionId{};
913 
914  for(size_t i = 0; i < sparseRegionIds.size(); ++i) {
915  sparseToDenseRegionId[sparseRegionIds[i]] = i;
916  }
917 
918  // update region id on all vertices: "sparse id" -> "dense id"
919 
920 #ifdef TTK_ENABLE_OPENMP
921 #pragma omp parallel for num_threads(threadNumber_)
922 #endif // TTK_ENABLE_OPENMP
923  for(size_t i = 0; i < nVerts; ++i) {
924  morseSmaleManifold[i] = sparseToDenseRegionId[morseSmaleManifold[i]];
925  }
926 
927  return 0;
