164  const std::vector<Cell> &criticalPoints,

167  const triangulationType &triangulation) const {
168 
169  std::vector<SimplexId> saddleIndexes;
170  const SimplexId numberOfCriticalPoints = criticalPoints.size();
171  for(SimplexId i = 0; i < numberOfCriticalPoints; ++i) {
172  const Cell &criticalPoint = criticalPoints[i];
173 
174  if(criticalPoint.dim_ == 1)
175  saddleIndexes.push_back(i);
176  }
177  const SimplexId numberOfSaddles = saddleIndexes.size();
178 
179  // estimation of the number of separatrices, apriori :
180  // numberOfAscendingPaths=2, numberOfDescendingPaths=2
181  const SimplexId numberOfSeparatrices = 4 * numberOfSaddles;
182  separatrices.resize(numberOfSeparatrices);
183  separatricesGeometry.resize(numberOfSeparatrices);
184 
185  // apriori: by default construction, the separatrices are not valid
186 #ifdef TTK_ENABLE_OPENMP
187 #pragma omp parallel for num_threads(threadNumber_)
188 #endif
189  for(SimplexId i = 0; i < numberOfSaddles; ++i) {
190  const SimplexId saddleIndex = saddleIndexes[i];
191  const Cell &saddle = criticalPoints[saddleIndex];
192 
193  // add ascending vpaths
194  {
195  const SimplexId starNumber = triangulation.getEdgeStarNumber(saddle.id_);
196  for(SimplexId j = 0; j < starNumber; ++j) {
197  const SimplexId shift = j;
198 
199  SimplexId triangleId;
200  triangulation.getEdgeStar(saddle.id_, j, triangleId);
201 
202  std::vector<Cell> vpath;
203  vpath.push_back(saddle);
204  discreteGradient_.getAscendingPath(
205  Cell(2, triangleId), vpath, triangulation);
206 
207  const Cell &lastCell = vpath.back();
208  if(lastCell.dim_ == 2 and discreteGradient_.isCellCritical(lastCell)) {
209  const SimplexId separatrixIndex = 4 * i + shift;
210 
211  separatricesGeometry[separatrixIndex] = std::move(vpath);
212  separatrices[separatrixIndex]
213  = Separatrix(true, saddle, lastCell, false, separatrixIndex);
214  }
215  }
216  }
217  }
218 
219  return 0;
