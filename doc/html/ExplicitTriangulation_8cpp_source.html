587 int ExplicitTriangulation::writeToFile(std::ofstream &stream) const {
588 
589  // 1. magic bytes (char *)
590  stream.write(this->magicBytes_, std::strlen(this->magicBytes_));
591  // 2. format version (unsigned long)
592  writeBin(stream, this->formatVersion_);
593  // 3. dimensionality (int)
594  const auto dim = this->getDimensionality();
595  writeBin(stream, dim);
596  // 4. number of vertices (SimplexId)
597  const auto nVerts = this->getNumberOfVertices();
598  writeBin(stream, nVerts);
599  // 5. number of edges (SimplexId)
600  const auto edgesNumber = [this, dim]() -> SimplexId {
601  if(dim == 1) {
602  return this->getNumberOfCells();
603  } else if(dim > 1) {
604  return this->edgeList_.size();
605  }
606  return 0;
607  };
608  const auto nEdges = edgesNumber();
609  writeBin(stream, nEdges);
610  // 6. number of triangles (SimplexId, 0 in 1D)
611  const auto trianglesNumber = [this, dim]() -> SimplexId {
612  if(dim == 2) {
613  return this->getNumberOfCells();
614  } else if(dim == 3) {
615  return this->triangleList_.size();
616  }
617  return 0;
618  };
619  const auto nTriangles = trianglesNumber();
620  writeBin(stream, nTriangles);
621  // 7. number of tetrahedron (SimplexId, 0 in 2D)
622  const auto nTetras = dim > 2 ? this->getNumberOfCells() : 0;
623  writeBin(stream, nTetras);
624 
625  // only write buffers oustside this->cellArray_ (cellVertex, vertexCoords),
626  // those ones will be provided by VTK
627 
628  // fixed-size arrays (in AbstractTriangulation.h)
629 
630 #define WRITE_FIXED(ARRAY) \
631  if(ARRAY.empty()) { \
632  writeBin(stream, char{0}); \
633  } else { \
634  writeBin(stream, char{1}); \
635  writeBinArray(stream, ARRAY.data(), ARRAY.size()); \
636  }
637 
638  // 8. edgeList (SimplexId array)
639  WRITE_FIXED(this->edgeList_);
640  // 9. triangleList (SimplexId array)
641  WRITE_FIXED(this->triangleList_);
642  // 10. triangleEdgeList (SimplexId array)
643  WRITE_FIXED(this->triangleEdgeList_);
644  // 11. tetraEdgeList (SimplexId array)
645  WRITE_FIXED(this->tetraEdgeList_);
646  // 12. tetraTriangleList (SimplexId array)
647  WRITE_FIXED(this->tetraTriangleList_);
648 
649  // variable-size arrays (FlatJaggedArray in ExplicitTriangulation.h)
650 
651 #define WRITE_GUARD(ARRAY) \
652  if(ARRAY.empty()) { \
653  writeBin(stream, char{0}); \
654  return; \
655  } else { \
656  writeBin(stream, char{1}); \
657  }
658 
659  const auto write_variable = [&stream](const FlatJaggedArray &arr) {
660  // empty array guard
661  WRITE_GUARD(arr);
662  writeBinArray(stream, arr.offset_ptr(), arr.subvectorsNumber() + 1);
663  writeBinArray(stream, arr.get_ptr(0, 0), arr.dataSize());
664  };
665 
666  // 13. vertexNeighbors (SimplexId array, offsets then data)
667  write_variable(this->vertexNeighborData_);
668  // 14. cellNeighbors (SimplexId array, offsets then data)
669  write_variable(this->cellNeighborData_);
670  // 15. vertexEdges (SimplexId array, offsets then data)
671  write_variable(this->vertexEdgeData_);
672  // 16. vertexTriangles (SimplexId array, offsets then data)
673  write_variable(this->vertexTriangleData_);
674  // 17. edgeTriangles (SimplexId array, offsets then data)
675  write_variable(this->edgeTriangleData_);
676  // 18. vertexStars (SimplexId array, offsets then data)
677  write_variable(this->vertexStarData_);
678  // 19. edgeStars (SimplexId array, offsets then data)
679  write_variable(this->edgeStarData_);
680  // 20. triangleStars (SimplexId array, offsets then data)
681  write_variable(this->triangleStarData_);
682  // 21. vertexLinks (SimplexId array, offsets then data)
683  write_variable(this->vertexLinkData_);
684  // 22. edgeLinks (SimplexId array, offsets then data)
685  write_variable(this->edgeLinkData_);
686  // 23. triangleLinks (SimplexId array, offsets then data)
687  write_variable(this->triangleLinkData_);
688 
689  const auto write_bool = [&stream](const std::vector<bool> &arr) {
690  // empty array guard
691  WRITE_GUARD(arr);
692  for(size_t i = 0; i < arr.size(); ++i) {
693  const auto b = static_cast<char>(arr[i]);
694  writeBin(stream, b);
695  }
696  };
697 
698  // 24. boundary vertices (bool array)
699  write_bool(this->boundaryVertices_);
700  // 25. boundary edges (bool array)
701  write_bool(this->boundaryEdges_);
702  // 26. boundary triangles (bool array)
703  write_bool(this->boundaryTriangles_);
704 
705  return 0;
706 }
