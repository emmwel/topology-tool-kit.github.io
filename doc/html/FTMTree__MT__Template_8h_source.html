347  std::tuple<bool, bool> FTMTree_MT::propage(const triangulationType *mesh,
348  CurrentState &currentState,
349  UF curUF) {
350  bool becameSaddle = false, isLast = false;
351  const auto nbNeigh = mesh->getVertexNeighborNumber(currentState.vertex);
352  valence decr = 0;
353 
354  // once for all
355  auto *curUFF = curUF->find();
356 
357  // propagation / is saddle
358  for(valence n = 0; n < nbNeigh; ++n) {
359  SimplexId neigh;
360  mesh->getVertexNeighbor(currentState.vertex, n, neigh);
361 
362  if(comp_.vertLower(neigh, currentState.vertex)) {
363  UF neighUF = (*mt_data_.ufs)[neigh];
364 
365  // is saddle
366  if(!neighUF || neighUF->find() != curUFF) {
367  becameSaddle = true;
368  } else if(neighUF) {
369  ++decr;
370  }
371 
372  } else {
373  if(!(*mt_data_.propagation)[neigh]
374  || (*mt_data_.propagation)[neigh]->find() != curUFF) {
375  currentState.addNewVertex(neigh);
376  (*mt_data_.propagation)[neigh] = curUFF;
377  }
378  }
379  }
380 
381  // is last
382  valence oldVal;
383  valence &tmp = (*mt_data_.valences)[currentState.vertex];
384 #ifdef TTK_ENABLE_OPENMP
385 #pragma omp atomic capture
386 #endif
387  {
388  oldVal = tmp;
389  tmp -= decr;
390  }
391  if(oldVal == decr) {
392  isLast = true;
393  }
394 
395  return std::make_tuple(becameSaddle, isLast);
396  }
