171  const auto point = &coords[p * 3];
172  _coords.push_back(point[0]);
173  _coords.push_back(point[1]);
174  _coords.push_back(point[2]);
175  _scalars.push_back(scalarBuf[p]);
176  _isovals.push_back(isoval);
177  _flags.push_back(code == minCode ? 0 : 1);
178  };

180  const vtkIdType nc = arcs->GetNumberOfCells();
181  for(vtkIdType c = 0; c < nc; ++c) {
182  const auto p = c2p[c];
183  const auto q = c2q[c];
184  const auto pCode = codeBuf[p];
185  const auto qCode = codeBuf[q];
186 
187  const bool pIsSad = pCode != minCode && pCode != maxCode;
188  const bool qIsSad = qCode != minCode && qCode != maxCode;
189  if(pIsSad || qIsSad) {
190  if(pIsSad && qIsSad) // two saddles
191  continue;
192  // extremum and saddle
193  const auto ext = pIsSad ? q : p;
194  const auto sad = pIsSad ? p : q;
195  const float isoval = scalarBuf[ext] * extFac + scalarBuf[sad] * sadFac;
196  addPoint(ext, isoval, codeBuf[ext]);
197  } else { // min-max pair

199  const auto pVal = scalarBuf[p];
200  const auto qVal = scalarBuf[q];
201  const auto cVal = (pVal + qVal) / 2;
202  addPoint(p, pVal * extFac + cVal * sadFac, pCode);
203  addPoint(q, qVal * extFac + cVal * sadFac, qCode);
204  }
205  }
