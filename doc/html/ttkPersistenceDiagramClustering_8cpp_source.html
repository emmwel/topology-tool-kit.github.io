532  vtkNew<vtkTransformFilter> trf{};
533  trf->SetTransform(tr);
534  trf->SetInputData(vtu);
535  trf->Update();
536  output->SetBlock(i, trf->GetOutputDataObject(0));
537 
538  } else {
539  // add diagram to output multi-block dataset
540  output->SetBlock(i, vtu);
541  }
542  }
