665  std::vector<KDTree<dataType> *> &correspondance_kdt_map,
666  std::priority_queue<std::pair<int, dataType>,
667  std::vector<std::pair<int, dataType>>,
668  Compare<dataType>> &diagonal_queue,
669  const int kdt_index) {
671 
672  // First, find the lowest and second lowest weights for diagonal goods
673  // Take this time to update the weights in the priority queue of the goods
674  // tested. It is not necessary to update weights for all diagonal bidders in
675  // the pririty queue since weights can only increase and we are interested
676  // only in the lowest ones
677  bool updated_top_pair
678  = false; // Boolean which equals true iff the top pair in the priority
679  // queue is given the good price
680  bool non_empty_goods = goods->size() > 0;
681  std::pair<int, dataType> best_pair;
682  if(non_empty_goods) {
683  while(!updated_top_pair) {
684  std::pair<int, dataType> top_pair = diagonal_queue.top();
685  dataType queue_weight = top_pair.second;
686  Good<dataType> *good = &(goods->get(top_pair.first));
687 
688  diagonal_queue.pop();
689  if(good->getPrice() > queue_weight) {
690  // If the weight in the priority queue is not the good one, update it
691  std::get<1>(top_pair) = good->getPrice();
692  diagonal_queue.push(top_pair);
693  } else {
694  updated_top_pair = true;
695  best_pair = top_pair;
696  }
697  }
698  }
699  // std::cout << "got here 1" << std::endl;
700  std::pair<int, dataType> second_pair;
701  if(non_empty_goods && diagonal_queue.size() > 0) {
702  bool updated_second_pair = false;
703  while(!updated_second_pair) {
704  second_pair = diagonal_queue.top();
705  dataType queue_weight = second_pair.second;
706  Good<dataType> *good = &(goods->get(second_pair.first));
707  if(good->getPrice() > queue_weight) {
708  // If the weight in the priority queue is not the good one, update it
709  diagonal_queue.pop();
710  std::get<1>(second_pair) = good->getPrice();
711  diagonal_queue.push(second_pair);
712  } else {
713  updated_second_pair = true;
714  }
715  }
716  }
717 
718  // std::cout << "got here 1/2" << std::endl;
719  Good<dataType> *best_good{};
720  dataType best_val = 0;
721  dataType second_val = 0;
722  if(non_empty_goods) {
723  best_val = -best_pair.second;
724  second_val = diagonal_queue.size() > 0 ? -second_pair.second : best_val;
725  best_good = &(goods->get(best_pair.first));
726  }
727  // std::cout << "got here 2" << std::endl;
728 
729  // And now check for the corresponding twin bidder
730  bool is_twin = false;
731  Good<dataType> &g = twinGood;
732  dataType val = -this->cost(g, wasserstein, geometricalFactor);
733  val -= g.getPrice();
734  if(non_empty_goods) {
735  if(val > best_val) {
736  second_val = best_val;
737  best_val = val;
738  best_good = &g;
739  is_twin = true;
740  } else if(val > second_val || diagonal_queue.empty()) {
741  second_val = val;
742  }
743  } else {
744  best_val = val;
745  best_good = &g;
746  is_twin = true;
747  second_val = best_val;
748  }
749 
750  // std::cout << "got here 3" << std::endl;
751  if(second_val == std::numeric_limits<dataType>::lowest()) {
752  // There is only one acceptable good for the bidder
753  second_val = best_val;
754  }
755  dataType old_price = best_good->getPrice();
756  dataType new_price = old_price + best_val - second_val + epsilon;
757  if(new_price > std::numeric_limits<dataType>::max() / 2) {
758  new_price = old_price + epsilon;
759  std::cout << "Huho 592" << std::endl;
760  }
761 
762  // Assign bidder to best_good
763  this->setProperty(*best_good);
764  this->setPricePaid(new_price);
765 
766  // std::cout << "got here 4" << std::endl;
767  // Assign best_good to bidder and unassign the previous owner of best_good
768  // if need be
769  int idx_reassigned = best_good->getOwner();
770  best_good->assign(this->position_in_auction_, new_price);
771  if(is_twin) {
772  // std::cout << "got here 5" << std::endl;
773  // Update weight in KDTree if the closest good is in it
774  correspondance_kdt_map[best_good->id_]->updateWeight(
775  new_price, kdt_index);
776  if(non_empty_goods) {
777  diagonal_queue.push(best_pair);
778  }
779  } else {
780  // std::cout << "got here 6" << std::endl;
781  if(non_empty_goods) {
782  std::get<1>(best_pair) = new_price;
783  diagonal_queue.push(best_pair);
784  }
785  }

787  return idx_reassigned;
