690  const auto &nbDown = getNode(n)->getNumberOfDownSuperArcs();

701  const SuperArc *firstSA = getSuperArc(firstSaId);

703 
704  ExtendedUnionFind *merge = vect_SplitUF[firstChildNodeId]->find();
705  SimplexId further = merge->getOrigin();
706  idSuperArc furtherI = 0;
707 
708  for(idSuperArc ni = 1; ni < nbUp; ++ni) {
709  // find the more persistant way
710  const idSuperArc curSaId = getNode(n)->getUpSuperArcId(ni);
711  const SuperArc *curSA = getSuperArc(curSaId);
712  // Ignore hidden / simplified arc
713  if(!curSA->isVisible())
714  continue;
715  const idNode &neigh = curSA->getUpNodeId();
716 
717  // fix
718  if(neigh == n)
719  continue;
720 
721  // std::cout << "visit neighbor : " << ni << " which is " <<
722  // getNode(neigh)->getVertexId() << std::endl;
723 
724  ExtendedUnionFind *neighUF = vect_SplitUF[neigh]->find();
725 
726  if(isHigher(neighUF->getOrigin(), further)) {
727  further = neighUF->getOrigin();
728  furtherI = ni;
729  }
730  }
731 
732  if(nbUp > 1) {
733  // close finsh pair and make union
734  for(idSuperArc ni = 0; ni < nbUp; ++ni) {
735  const idSuperArc curSaId = getNode(n)->getUpSuperArcId(ni);
736  const SuperArc *curSA = getSuperArc(curSaId);
737  const idNode &neigh = curSA->getUpNodeId();
738 
739  // fix
740  if(neigh == n)
741  continue;
742 
743  ExtendedUnionFind *neighUF = vect_SplitUF[neigh]->find();
744 
745  if(ni != furtherI) {
746  addPair<scalarType>(
747  pairsST, neighUF->getOrigin(), v, mesh, false);
748 
749  pendingMinMax.erase(neighUF->getOrigin());
750 
751  // std::cout << " st make pair : " <<
752  // neighUF->getOrigin() << " - " << v
753  //<< " for neighbor " <<
754  // getNode(neigh)->getVertexId() << std::endl;
755  }
756 
757  ExtendedUnionFind::makeUnion(merge, neighUF)
758  ->setOrigin(further);
759  // Re-visit after merge lead to add the most persistant
760  // pair....
761  }
762  }
763  merge->find()->setOrigin(further);
764  vect_SplitUF[n] = merge->find();
765 
766  if(!nbDown) {
767  pendingMinMax[further] = v;
768  }
769 
770  } // end nbUp == 0 else
771  } // end for each node
772 
773  // Add the pending biggest pair of each component
774  for(const auto &pair_vert : pendingMinMax) {
775  addPair<scalarType>(
776  pairsST, pair_vert.first, pair_vert.second, mesh, false);
777  }
778  } // end para section
779  } // end para

1005  void MergeTree::processVertex(const SimplexId &currentVertex,
1006  std::vector<ExtendedUnionFind *> &vect_baseUF,
1007  const bool overlapB,
1008  const bool overlapA,
1009  const triangulationType &mesh,
1010  DebugTimer &begin) {

1018

1020  idNode closingNode, currentNode;
1021  SimplexId neighbor;
1022 
1023  // Check UF in neighborhood
1024  for(SimplexId n = 0; n < neighborNumber; ++n) {
1025  mesh->getVertexNeighbor(currentVertex, n, neighbor);
1026  // if the vertex is out: consider it null
1027  tmpseed = vect_baseUF[neighbor];

1029  if(tmpseed == nullptr) {
1030  continue;
1031  }
1032 
1033  tmpseed = tmpseed->find();
1034 
1035  // get all different UF in neighborhood
1036  if(find(vect_neighUF.cbegin(), vect_neighUF.cend(), tmpseed)
1037  == vect_neighUF.end()) {
1038  vect_neighUF.emplace_back(tmpseed);
1039  seed = tmpseed;

1041  }
1042 
1043  (neighSize = vect_neighUF.size());
1044 
1045  // SimplexId test = 1;

1047  // cout << test << " : " << vect_neighUF.size() << " " <<
1048  // vect_interfaceUF.size() << endl; Make output

1050  // we are on a real extrema we have to create a new UNION FIND and a
1051  // branch a real extrema can't be a virtual extrema
1052 
1053  seed = new ExtendedUnionFind(currentVertex);
1054  // When creating an extrema we create a pair ending on this node.
1055  currentNode = makeNode(currentVertex);
1056  getNode(currentNode)->setOrigin(currentNode);
1057  currentArc = openSuperArc(currentNode, overlapB, overlapA);
1058  // if(overlap && partition_ == 1) cout << currentVertex << endl;
1059  treeData_.leaves.emplace_back(currentNode);
1060 
1061  if(params_->debugLevel >= advancedInfoMsg) {
1062  if(isJT) {
1063  this->printMsg("Min node id: " + std::to_string(currentVertex), 1.0,
1064  begin.getElapsedTime(), this->threadNumber_);
1065  } else {
1066  this->printMsg("Max node id: " + std::to_string(currentVertex), 1.0,
1067  begin.getElapsedTime(), this->threadNumber_);
1068  }
1069  }
1070  } else if(neighSize > 1) {
1071  // Is a saddle if have more than one UF in neighborhood
1072  // Or is linked to an interface (virtual extrema -> leaves or
1073  // interface cc representant : regular node )
1074 
1075  // Merge operation => close all arriving SuperArc (ignoring UF from
1076  // interface) And open a new one
1077  closingNode = makeNode(currentVertex);
1078  currentArc = openSuperArc(closingNode, overlapB, overlapA);
1079 
1080  SimplexId farOrigin = vect_neighUF[0]->find()->getOrigin();
1081 
1082  // close each SuperArc finishing here
1083  for(auto *neigh : vect_neighUF) {
1084  closeSuperArc((idSuperArc)neigh->find()->getData(), closingNode,
1085  overlapB, overlapA);
1086  // persistance pair closing here.
1087  // For the one who will continue, it will be overide later
1088  vertex2Node(neigh->find()->getOrigin())->setTerminaison(closingNode);
1089 
1090  // cout <<
1091  // getNode(getCorrespondingNode(neigh->find()->getOrigin()))->getVertexId()
1092  //<< " terminate on " << getNode(closingNode)->getVertexId() << endl;
1093 
1094  if((isJT && isLower(neigh->find()->getOrigin(), farOrigin))
1095  || (!isJT && isHigher(neigh->find()->getOrigin(), farOrigin))) {
1096  // here we keep the continuing the most persitant pair.
1097  // It means a pair end when a parent have another origin thant the
1098  // current leaf (or is the root) It might be not intuitive but it is
1099  // more convenient for degenerate cases
1100  farOrigin = neigh->find()->getOrigin();
1101  // cout << "find origin " << farOrigin << " for " << currentVertex
1102  // << " " << isJT
1103  //<< endl;
1104  }
1105  }
1106 
1107  // Union correspond to the merge
1108  seed = ExtendedUnionFind::makeUnion(vect_neighUF);
1109  if(seed == nullptr) {
1110  return;
1111  }
1112  seed->setOrigin(farOrigin);
1113  getNode(closingNode)->setOrigin(getCorrespondingNodeId(farOrigin));
1114 
1115  // cout << " " << getNode(closingNode)->getVertexId() << " have origin
1116  // at "
1117  //<< getNode(getCorrespondingNode(farOrigin))->getVertexId() << endl;
1118 
1119  if(params_->debugLevel >= advancedInfoMsg) {
1120  this->printMsg("Saddle node id: " + std::to_string(currentVertex));
1121  }
1122 
1123  } else {
1124 #ifndef TTK_ENABLE_KAMIKAZE
1125  if(seed == nullptr) {
1126  return;
1127  }
1128 #endif // TTK_ENABLE_KAMIKAZE
1129  // regular node
1130  currentArc = (idSuperArc)seed->find()->getData();
1131  updateCorrespondingArc(currentVertex, currentArc);
1132  }
1133  // common
1134  seed->setData((ufDataType)currentArc);
1135  getSuperArc(currentArc)->setLastVisited(currentVertex);
1136  vect_baseUF[currentVertex] = seed;
1137  }
