254 std::pair<ttk::SimplexId, ttk::SimplexId>
255  ttk::ScalarFieldCriticalPoints::getNumberOfLowerUpperComponents(
256  const SimplexId vertexId,
257  const SimplexId *const offsets,
258  const triangulationType *triangulation) const {
259 
260  SimplexId neighborNumber = triangulation->getVertexNeighborNumber(vertexId);
261  std::vector<SimplexId> lowerNeighbors, upperNeighbors;
262 
263  for(SimplexId i = 0; i < neighborNumber; i++) {
264  SimplexId neighborId = 0;
265  triangulation->getVertexNeighbor(vertexId, i, neighborId);
266 
267  if(offsets[neighborId] < offsets[vertexId]) {
268  lowerNeighbors.push_back(neighborId);
269  }
270 
271  // upper link
272  if(offsets[neighborId] > offsets[vertexId]) {
273  upperNeighbors.push_back(neighborId);
274  }
275  }
276 
277  // shortcut, if min or max do not construct the complete star
278  if(!forceNonManifoldCheck && lowerNeighbors.empty()) {
279  // minimum
280  return std::make_pair(0, 1);
281  }
282 
283  if(!forceNonManifoldCheck && upperNeighbors.empty()) {
284  // maximum
285  return std::make_pair(1, 0);
286  }
287 
288  // now do the actual work
289  std::vector<UnionFind> lowerSeeds(lowerNeighbors.size());
290  std::vector<UnionFind *> lowerList(lowerNeighbors.size());
291  std::vector<UnionFind> upperSeeds(upperNeighbors.size());
292  std::vector<UnionFind *> upperList(upperNeighbors.size());
293 
294  for(SimplexId i = 0; i < (SimplexId)lowerSeeds.size(); i++) {
295  lowerList[i] = &(lowerSeeds[i]);
296  }
297  for(SimplexId i = 0; i < (SimplexId)upperSeeds.size(); i++) {
298  upperList[i] = &(upperSeeds[i]);
299  }
300 
301  SimplexId vertexStarSize = triangulation->getVertexStarNumber(vertexId);
302 
303  for(SimplexId i = 0; i < vertexStarSize; i++) {
304  SimplexId cellId = 0;
305  triangulation->getVertexStar(vertexId, i, cellId);
306 
307  SimplexId cellSize = triangulation->getCellVertexNumber(cellId);
308  for(SimplexId j = 0; j < cellSize; j++) {
309  SimplexId neighborId0 = -1;
310  triangulation->getCellVertex(cellId, j, neighborId0);
311 
312  if(neighborId0 != vertexId) {
313  // we are on the link
314 
315  bool lower0 = offsets[neighborId0] < offsets[vertexId];
316 
317  // connect it to everybody except himself and vertexId
318  for(SimplexId k = j + 1; k < cellSize; k++) {
319 
320  SimplexId neighborId1 = -1;
321  triangulation->getCellVertex(cellId, k, neighborId1);
322 
323  if((neighborId1 != neighborId0) && (neighborId1 != vertexId)) {
324 
325  bool lower1 = offsets[neighborId1] < offsets[vertexId];
326 
327  std::vector<SimplexId> *neighbors = &lowerNeighbors;
328  std::vector<UnionFind *> *seeds = &lowerList;
329 
330  if(!lower0) {
331  neighbors = &upperNeighbors;
332  seeds = &upperList;
333  }
334 
335  if(lower0 == lower1) {
336  // connect their union-find sets!
337  SimplexId lowerId0 = -1, lowerId1 = -1;
338  for(SimplexId l = 0; l < (SimplexId)neighbors->size(); l++) {
339  if((*neighbors)[l] == neighborId0) {
340  lowerId0 = l;
341  }
342  if((*neighbors)[l] == neighborId1) {
343  lowerId1 = l;
344  }
345  }
346  if((lowerId0 != -1) && (lowerId1 != -1)) {
347  (*seeds)[lowerId0]
348  = makeUnion((*seeds)[lowerId0], (*seeds)[lowerId1]);
349  (*seeds)[lowerId1] = (*seeds)[lowerId0];
350  }
351  }
352  }
353  }
354  }
355  }
356  }
357 
358  // let's remove duplicates now
359 
360  // update the UF if necessary
361  for(SimplexId i = 0; i < (SimplexId)lowerList.size(); i++)
362  lowerList[i] = lowerList[i]->find();
363  for(SimplexId i = 0; i < (SimplexId)upperList.size(); i++)
364  upperList[i] = upperList[i]->find();
365 
366  std::vector<UnionFind *>::iterator it;
367  std::sort(lowerList.begin(), lowerList.end());
368  it = unique(lowerList.begin(), lowerList.end());
369  lowerList.resize(distance(lowerList.begin(), it));
370 
371  std::sort(upperList.begin(), upperList.end());
372  it = unique(upperList.begin(), upperList.end());
373  upperList.resize(distance(upperList.begin(), it));
374 
375  if(debugLevel_ >= (int)(debug::Priority::VERBOSE)) {
376  printMsg("Vertex #" + std::to_string(vertexId)
377  + ": lowerLink-#CC=" + std::to_string(lowerList.size())
378  + " upperLink-#CC=" + std::to_string(upperList.size()),
379  debug::Priority::VERBOSE);
380  }
381 
382  return std::make_pair(lowerList.size(), upperList.size());
383 }

385 template <class triangulationType>
386 char ttk::ScalarFieldCriticalPoints::getCriticalType(
387  const SimplexId &vertexId,
388  const SimplexId *const offsets,
389  const triangulationType *triangulation) const {
390 
391  SimplexId downValence, upValence;
392  std::tie(downValence, upValence)
393  = getNumberOfLowerUpperComponents(vertexId, offsets, triangulation);
394 
395  if(downValence == 0 && upValence == 1) {
396  return (char)(CriticalType::Local_minimum);
397  } else if(downValence == 1 && upValence == 0) {
398  return (char)(CriticalType::Local_maximum);
399  } else if(downValence == 1 && upValence == 1) {
400  // regular point
401  return (char)(CriticalType::Regular);
402  } else {
403  // saddles
404  if(dimension_ == 2) {
405  if((downValence == 2 && upValence == 1)
406  || (downValence == 1 && upValence == 2)
407  || (downValence == 2 && upValence == 2)) {
408  // regular saddle
409  return (char)(CriticalType::Saddle1);
410  } else {
411  // monkey saddle, saddle + extremum
412  return (char)(CriticalType::Degenerate);
413  // NOTE: you may have multi-saddles on the boundary in that
414  // configuration
415  // to make this computation 100% correct, one would need to
416  // disambiguate boundary from interior vertices
417  }
418  } else if(dimension_ == 3) {
419  if(downValence == 2 && upValence == 1) {
420  return (char)(CriticalType::Saddle1);
421  } else if(downValence == 1 && upValence == 2) {
422  return (char)(CriticalType::Saddle2);
423  } else {
424  // monkey saddle, saddle + extremum
425  return (char)(CriticalType::Degenerate);
426  // NOTE: we may have a similar effect in 3D (TODO)
427  }
428  }
429  }
430 
431  // -2: regular points
432  return (char)(CriticalType::Regular);
433 }
