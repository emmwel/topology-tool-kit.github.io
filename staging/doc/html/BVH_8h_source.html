123  const Triangle &t = triangles[i];
124  minX = std::min(t.m_minX, minX);
125  minY = std::min(t.m_minY, minY);
126  minZ = std::min(t.m_minZ, minZ);
127 
128  maxX = std::max(t.m_maxX, maxX);
129  maxY = std::max(t.m_maxY, maxY);
130  maxZ = std::max(t.m_maxZ, maxZ);
131  }
132  int numberTriangles = end - start;
133  if(numberTriangles == 1) {
134  const Triangle &t = triangles[start];
135  std::vector<int> indices = {t.m_index};
136  float pMin[3] = {t.m_minX, t.m_minY, t.m_minZ};
137  float pMax[3] = {t.m_maxX, t.m_maxY, t.m_maxZ};
138  return new Node(indices, 1, pMin, pMax);
139  } else {
140  // find the bounds of the centroids, figure out what dimension to split
141  // on
142  float cminX, cminY, cminZ;
143  float cmaxX, cmaxY, cmaxZ;
144  cminX = cminY = cminZ = std::numeric_limits<float>::max();
145  cmaxX = cmaxY = cmaxZ = std::numeric_limits<float>::min();
146  for(size_t i = start; i < end; i++) {
147  const Triangle &t = triangles[i];
148  cminX = std::min(t.m_centroid_x, cminX);
149  cminY = std::min(t.m_centroid_y, cminY);
150  cminZ = std::min(t.m_centroid_z, cminZ);
151 
152  cmaxX = std::max(t.m_centroid_x, cmaxX);
153  cmaxY = std::max(t.m_centroid_y, cmaxY);
154  cmaxZ = std::max(t.m_centroid_z, cmaxZ);
155  }
156  // figure out the biggest extent, use that dimension
157  float diffX = std::abs(cmaxX - cminX);
158  float diffY = std::abs(cmaxY - cminY);
159  float diffZ = std::abs(cmaxZ - cminZ);
160  float maximumExtent = std::max({diffX, diffY, diffZ});
161  int axis;
162  float minToCheck, maxToCheck;
163  if(maximumExtent == diffX) {
164  axis = 0;
165  minToCheck = cminX;
166  maxToCheck = cmaxX;
167  } else if(maximumExtent == diffY) {
168  axis = 1;
169  minToCheck = cminY;
170  maxToCheck = cmaxY;
171  } else {
172  axis = 2;
173  minToCheck = cminZ;
174  maxToCheck = cmaxZ;
175  }
176  size_t half = (start + end) / 2;
177  // partition triangles into two sets and build children
178  if(minToCheck == maxToCheck) {
179 
180  std::vector<int> triangleIndices;
181  for(size_t i = start; i < end; i++) {
182  triangleIndices.push_back(triangles[i].m_index);
183  }
184  float pMin[3] = {minX, minY, minZ};
185  float pMax[3] = {maxX, maxY, maxZ};

187  } else {

255

257 
258  CinemaImaging::subVectors(v0v1, &vertexCoords[v0], &vertexCoords[v1]);
259  CinemaImaging::subVectors(v0v2, &vertexCoords[v0], &vertexCoords[v2]);
260 
261  CinemaImaging::cross(pvec, ray.m_direction, v0v2);
262  float det = CinemaImaging::dot(v0v1, pvec);
263  if(det > -kEpsilon && det < kEpsilon)
264  return false;
265 
266  float invDet = 1.0f / det;
267 
268  CinemaImaging::subVectors(tvec, &vertexCoords[v0], ray.m_origin);
269  float u = CinemaImaging::dot(tvec, pvec) * invDet;
270  if(u < 0.0 || u > 1.0)
271  return false;
272 
273  CinemaImaging::cross(qvec, tvec, v0v1);
274  float v = CinemaImaging::dot(ray.m_direction, qvec) * invDet;
275  if(v < 0.0 || u + v > 1.0)
276  return false;
277 
278  float t = CinemaImaging::dot(v0v2, qvec) * invDet;
279  ray.distance = t;
280 
281  ray.u = u;
282  ray.v = v;
283 
284  return true;
