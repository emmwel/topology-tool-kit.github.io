602  && (boundaryEdges_.size() == edgeList_.size())) {
603  return 0;
604  }
605 
606  preconditionEdgesInternal();
607  boundaryEdges_.resize(edgeList_.size(), false);
608 
609  if(getDimensionality() == 2) {
610  preconditionEdgeStarsInternal();
611  for(SimplexId i = 0; i < (SimplexId)edgeStarList_.size(); i++) {
612  if(edgeStarList_[i].size() == 1) {

666  inline int preconditionBoundaryVerticesInternal() override {

669  && ((SimplexId)boundaryVertices_.size() == vertexNumber_))
670  return 0;
671 
672  boundaryVertices_.resize(vertexNumber_, false);
673 
674  // create the list of boundary elements
675  // create their star
676  // look for singletons
677  if(getDimensionality() == 1) {
678  preconditionVertexStarsInternal();
679  for(SimplexId i = 0; i < (SimplexId)vertexStarList_.size(); i++) {
680  if(vertexStarList_[i].size() == 1) {
681  boundaryVertices_[i] = true;
682  }
683  }
684  } else if(getDimensionality() == 2) {
685  preconditionEdgesInternal();
686  preconditionEdgeStarsInternal();
687 
688  for(SimplexId i = 0; i < (SimplexId)edgeStarList_.size(); i++) {
689  if(edgeStarList_[i].size() == 1) {
690  boundaryVertices_[edgeList_[i].first] = true;
691  boundaryVertices_[edgeList_[i].second] = true;

701  boundaryVertices_[triangleList_[i][1]] = true;
702  boundaryVertices_[triangleList_[i][2]] = true;
703  }
704  }
