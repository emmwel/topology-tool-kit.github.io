128 void FTMTree_MT::buildSegmentation() {
129 
130  const idSuperArc nbArcs = mt_data_.superArcs->size();
131 
132  // Make reserve
133 
134  // SuperArc i correspond to segment i,
135  // one arc correspond to one segment
136  vector<SimplexId> sizes(nbArcs);
137 
138  // get the size of each segment
139  const idSuperArc arcChunkSize = getChunkSize(nbArcs);
140  const idSuperArc arcChunkNb = getChunkCount(nbArcs);
141  for(idSuperArc arcChunkId = 0; arcChunkId < arcChunkNb; ++arcChunkId) {
142  // WHY shared(sizes) is needed ??
143 #ifdef TTK_ENABLE_OPENMP
144 #pragma omp task firstprivate(arcChunkId) shared(sizes) \
145  OPTIONAL_PRIORITY(isPrior())
146 #endif
147  {
148  const idSuperArc lowerBound = arcChunkId * arcChunkSize;
149  const idSuperArc upperBound
150  = min(nbArcs, (arcChunkId + 1) * arcChunkSize);
151  for(idSuperArc a = lowerBound; a < upperBound; ++a) {
152  sizes[a]
153  = max(SimplexId{0}, (*mt_data_.superArcs)[a].getNbVertSeen() - 1);
154  }
155  }
156  }
157 #ifdef TTK_ENABLE_OPENMP
158 #pragma omp taskwait
159 #endif
160 
161  // change segments size using the created vector
162  mt_data_.segments_.resize(sizes);
163 
164  Timer segmentsSet;
165 
166  // Fill segments using vert2tree
167 
168  // current status of the segmentation of this arc
169  vector<SimplexId> posSegm(nbArcs, 0);
170 
171  // Segments are connex region of geometrie forming
172  // the segmentation (sorted in ascending order)
173  const SimplexId nbVert = scalars_->size;
174  const SimplexId chunkSize = getChunkSize();
175  const SimplexId chunkNb = getChunkCount();
176  for(SimplexId chunkId = 0; chunkId < chunkNb; ++chunkId) {
177 #ifdef TTK_ENABLE_OPENMP
178 #pragma omp task firstprivate(chunkId) shared(posSegm) \
179  OPTIONAL_PRIORITY(isPrior())
180 #endif
181  {
182  const SimplexId lowerBound = chunkId * chunkSize;
183  const SimplexId upperBound = min(nbVert, (chunkId + 1) * chunkSize);
184  for(SimplexId i = lowerBound; i < upperBound; ++i) {
185  const auto vert = scalars_->sortedVertices[i];
186  if(isCorrespondingArc(vert)) {
187  idSuperArc sa = getCorrespondingSuperArcId(vert);
188  SimplexId vertToAdd;
189  if((*mt_data_.visitOrder)[vert] != nullVertex) {
190  // Opposite order for Split Tree
191  vertToAdd = (*mt_data_.visitOrder)[vert];
192  if(isST())
193  vertToAdd = getSuperArc(sa)->getNbVertSeen() - vertToAdd - 2;
194  mt_data_.segments_[sa][vertToAdd] = vert;
195  } else if(mt_data_.trunkSegments->size() == 0) {
196  // MT computation
197 #ifdef TTK_ENABLE_OPENMP
198 #pragma omp atomic capture
199 #endif
200  vertToAdd = posSegm[sa]++;
201  mt_data_.segments_[sa][vertToAdd] = vert;
202  }
203 
204  } // end is arc
205  } // end for
206  } // end task
207  }
208 #ifdef TTK_ENABLE_OPENMP
209 #pragma omp taskwait
210 #endif
211 
212  printTime(segmentsSet, "segmentation set vertices", -1, 4);
213 
214  if(mt_data_.trunkSegments->size() == 0) {
215  // sort arc that have been filled by the trunk
216  // only for MT
217  Timer segmentsSortTime;
218  for(idSuperArc a = 0; a < nbArcs; ++a) {
219  if(posSegm[a]) {
220 #ifdef TTK_ENABLE_OPENMP
221 #pragma omp task firstprivate(a) OPTIONAL_PRIORITY(isPrior())
222 #endif
223  mt_data_.segments_[a].sort(scalars_);
224  }
225  }
226 #ifdef TTK_ENABLE_OPENMP
227 #pragma omp taskwait
228 #endif
229  printTime(segmentsSortTime, "segmentation sort vertices", -1, 4);
230  } else {
231  // Contour tree: we create the arc segmentation for arcs in the trunk
232  Timer segmentsArcTime;
233  for(idSuperArc a = 0; a < nbArcs; ++a) {
234  // CT computation, we have already the vert list
235  if((*mt_data_.trunkSegments)[a].size()) {
236 #ifdef TTK_ENABLE_OPENMP
237 #pragma omp task firstprivate(a) OPTIONAL_PRIORITY(isPrior())
238 #endif
239  mt_data_.segments_[a].createFromList(
240  scalars_, (*mt_data_.trunkSegments)[a],
241  mt_data_.treeType == TreeType::Split);
242  }
243  }
244 #ifdef TTK_ENABLE_OPENMP
245 #pragma omp taskwait
246 #endif
247 
248  printTime(segmentsArcTime, "segmentation arcs lists", -1, 4);
249  }
250 
251  // Update SuperArc region
252 
253  // ST have a segmentation wich is in the reverse-order of its build
254  // ST have a segmentation sorted in ascending order as JT
255  for(idSuperArc arcChunkId = 0; arcChunkId < arcChunkNb; ++arcChunkId) {
256 #ifdef TTK_ENABLE_OPENMP
257 #pragma omp task firstprivate(arcChunkId) OPTIONAL_PRIORITY(isPrior())
258 #endif
259  {
260  const idSuperArc lowerBound = arcChunkId * arcChunkSize;
261  const idSuperArc upperBound
262  = min(nbArcs, (arcChunkId + 1) * arcChunkSize);
263  for(idSuperArc a = lowerBound; a < upperBound; ++a) {
264  // avoid empty region
265  if(mt_data_.segments_[a].size()) {
266  (*mt_data_.superArcs)[a].concat(
267  mt_data_.segments_[a].begin(), mt_data_.segments_[a].end());
268  }
269  }
270  }
271  }
272 #ifdef TTK_ENABLE_OPENMP
273 #pragma omp taskwait
274 #endif

447 idSuperArc FTMTree_MT::insertNode(Node *node, const bool segm) {
448  // Normal insert : existing arc stay below inserted (JT example)
449  // * - <- upNodeId
450  // | \ | <- newSA
451  // | * <- newNodeId
452  // | | <- currentSA
453  // - - -
454  // already present
455  if(isCorrespondingNode(node->getVertexId())) {
456  Node *myNode = vertex2Node(node->getVertexId());
457  // If it has been hidden / replaced we need to re-make it
458  idSuperArc correspondingArcId = myNode->getUpSuperArcId(0);
459  updateCorrespondingArc(myNode->getVertexId(), correspondingArcId);
460  }
461 
462  idNode upNodeId, newNodeId;
463  idSuperArc currentSA, newSA;
464  SimplexId origin;
465 
466  // Create new node
467  currentSA = getCorrespondingSuperArcId(node->getVertexId());
468  upNodeId = (*mt_data_.superArcs)[currentSA].getUpNodeId();
469  origin = (*mt_data_.nodes)[(*mt_data_.superArcs)[currentSA].getDownNodeId()]
470  .getOrigin();
471  newNodeId = makeNode(node, origin);
472 
473  // Connectivity
474  // Insert only node inside the partition : created arc don t cross
475  newSA = makeSuperArc(newNodeId, upNodeId);
476 
477  (*mt_data_.superArcs)[currentSA].setUpNodeId(newNodeId);
478  (*mt_data_.nodes)[upNodeId].removeDownSuperArc(currentSA);
479  (*mt_data_.nodes)[newNodeId].addDownSuperArcId(currentSA);
480 
481  // cut the vertex list at the node position and
482  // give each arc its part.
483  if(segm) {
484  if(mt_data_.treeType == TreeType::Split) {
485  (*mt_data_.superArcs)[newSA].concat(
486  get<1>((*mt_data_.superArcs)[currentSA].splitBack(
487  node->getVertexId(), scalars_)));
488  } else {
489  (*mt_data_.superArcs)[newSA].concat(
490  get<1>((*mt_data_.superArcs)[currentSA].splitFront(
491  node->getVertexId(), scalars_)));
492  }
493  }
494 
495  return newSA;
496 }

551 void FTMTree_MT::normalizeIds(void) {
552  Timer normTime;
553  sortLeaves(true);
554 
555  auto getNodeParentArcNb
556  = [&](const idNode curNode, const bool goUp) -> idSuperArc {
557  if(goUp) {
558  return getNode(curNode)->getNumberOfUpSuperArcs();
559  }
560 
561  return getNode(curNode)->getNumberOfDownSuperArcs();
562  };
563 
564  auto getNodeParentArc
565  = [&](const idNode curNode, const bool goUp, idSuperArc i) -> idSuperArc {
566  if(goUp) {
567  return getNode(curNode)->getUpSuperArcId(i);
568  }
569 
570  return getNode(curNode)->getDownSuperArcId(i);
571  };
572 
573  auto getArcParentNode
574  = [&](const idSuperArc curArc, const bool goUp) -> idNode {
575  if(goUp) {
576  return getSuperArc(curArc)->getUpNodeId();
577  }
578 
579  return getSuperArc(curArc)->getDownNodeId();
580  };
581 
582  std::queue<tuple<idNode, bool>> q;
583  std::stack<tuple<idNode, bool>> qr;
584  for(const idNode n : *mt_data_.leaves) {
585  bool goUp = isJT() || isST() || getNode(n)->getNumberOfUpSuperArcs();
586  if(goUp)
587  q.emplace(make_tuple(n, goUp));
588  else
589  qr.emplace(make_tuple(n, goUp));
590  }
591 
592  while(!qr.empty()) {
593  q.emplace(qr.top());
594  qr.pop();
595  }
596 
597  // Normalized id
598  idSuperArc nIdMin = 0;
599  idSuperArc nIdMax = getNumberOfSuperArcs() - 1;
600 
601  vector<bool> seenUp(getNumberOfSuperArcs(), false);
602  vector<bool> seenDown(getNumberOfSuperArcs(), false);
603 
604  while(!q.empty()) {
605  bool goUp;
606  idNode curNodeId;
607  tie(curNodeId, goUp) = q.front();
608  q.pop();
609 
610  if(goUp)
611  sortUpArcs(curNodeId);
612  else
613  sortDownArcs(curNodeId);
614 
615  // Assign arc above
616  const idSuperArc nbArcParent = getNodeParentArcNb(curNodeId, goUp);
617  for(idSuperArc pid = 0; pid < nbArcParent; pid++) {
618  const idSuperArc currentArcId = getNodeParentArc(curNodeId, goUp, pid);
619  if(goUp) {
620  if(getSuperArc(currentArcId)->getNormalizedId() == nullSuperArc) {
621  getSuperArc(currentArcId)->setNormalizeIds(nIdMin++);
622  }
623  if(!seenUp[currentArcId]) {
624  q.emplace(make_tuple(getArcParentNode(currentArcId, goUp), goUp));
625  seenUp[currentArcId] = true;
626  }
627  } else {
628  if(getSuperArc(currentArcId)->getNormalizedId() == nullSuperArc) {
629  getSuperArc(currentArcId)->setNormalizeIds(nIdMax--);
630  }
631  if(!seenDown[currentArcId]) {
632  q.emplace(make_tuple(getArcParentNode(currentArcId, goUp), goUp));
633  seenDown[currentArcId] = true;
634  }
635  }
636  }
637  }
638 
639 #ifndef TTK_ENABLE_KAMIKAZE
640  if(std::abs((long)nIdMax - (long)nIdMin) > 1) {
641  this->printMsg({"error during normalize, tree compromized: ",
642  std::to_string(nIdMin), " ", std::to_string(nIdMax)},
643  debug::Priority::ERROR);
644  }
645 #endif
646 
647  printTime(normTime, "normalize ids", -1, 4);
648 }

861 SimplexId FTMTree_MT::trunkCTSegmentation(const vector<SimplexId> &trunkVerts,
862  const SimplexId begin,
863  const SimplexId stop) {
864  const int nbTasksThreads = 40;
865  const auto sizeBackBone = abs(stop - begin);
866  const auto chunkSize = getChunkSize(sizeBackBone, nbTasksThreads);
867  const auto chunkNb = getChunkCount(sizeBackBone, nbTasksThreads);
868  // si pas efficace vecteur de la taille de node ici a la place de acc
869  idNode lastVertInRange = 0;
870  mt_data_.trunkSegments->resize(getNumberOfSuperArcs());
871  for(SimplexId chunkId = 0; chunkId < chunkNb; ++chunkId) {
872 #ifdef TTK_ENABLE_OPENMP
873 #pragma omp task firstprivate(chunkId, lastVertInRange) shared(trunkVerts) \
874  OPTIONAL_PRIORITY(isPrior())
875 #endif
876  {
877  vector<SimplexId> regularList;
878  if(params_->segm) {
879  regularList.reserve(25);
880  }
881  const SimplexId lowerBound = begin + chunkId * chunkSize;
882  const SimplexId upperBound
883  = min(stop, (begin + (chunkId + 1) * chunkSize));
884  if(lowerBound != upperBound) {
885  const SimplexId pos = isST() ? upperBound - 1 : lowerBound;
886  lastVertInRange
887  = getVertInRange(trunkVerts, scalars_->sortedVertices[pos], 0);
888  }
889  for(SimplexId v = lowerBound; v < upperBound; ++v) {
890  const SimplexId s
891  = isST() ? scalars_->sortedVertices[lowerBound + upperBound - 1 - v]
892  : scalars_->sortedVertices[v];
893  if(isCorrespondingNull(s)) {
894  const idNode oldVertInRange = lastVertInRange;
895  lastVertInRange = getVertInRange(trunkVerts, s, lastVertInRange);
896  const idSuperArc thisArc = upArcFromVert(trunkVerts[lastVertInRange]);
897  updateCorrespondingArc(s, thisArc);
898 
899  if(params_->segm) {
900  if(oldVertInRange == lastVertInRange) {
901  regularList.emplace_back(s);
902  } else {
903  // accumulated to have only one atomic update when needed
904  const idSuperArc oldArc
905  = upArcFromVert(trunkVerts[oldVertInRange]);
906  if(regularList.size()) {
907 #ifdef TTK_ENABLE_OPENMP
908 #pragma omp critical
909 #endif
910  {
911  (*mt_data_.trunkSegments)[oldArc].emplace_back(regularList);
912  regularList.clear();
913  }
914  }
915  // hand.vtu, sequential: 28554
916  regularList.emplace_back(s);
917  }
918  }
919  }
920  }
921  // force increment last arc
922  const idNode baseNode
923  = getCorrespondingNodeId(trunkVerts[lastVertInRange]);
924  const idSuperArc upArc = getNode(baseNode)->getUpSuperArcId(0);
925  if(regularList.size()) {
926 #ifdef TTK_ENABLE_OPENMP
927 #pragma omp critical
928 #endif
929  {
930  (*mt_data_.trunkSegments)[upArc].emplace_back(regularList);
931  regularList.clear();
932  }
933  }
934  }
935  }
936 #ifdef TTK_ENABLE_OPENMP
937 #pragma omp taskwait
938 #endif
939  // count added
940  SimplexId tot = 0;
941 #ifdef TTK_ENABLE_FTM_TREE_PROCESS_SPEED
942  for(const auto &l : *mt_data_.trunkSegments) {
943  SimplexId arcSize = 0;
944  for(const auto &v : l) {
945  arcSize += v.size();
946  }
947  tot += arcSize;
948  }
949 #endif
950  return tot;
951 }

953 SimplexId FTMTree_MT::trunkSegmentation(const vector<SimplexId> &trunkVerts,
954  const SimplexId begin,
955  const SimplexId stop) {
956  // Assign missing vert to the good arc
957  // and also add the corresponding number for
958  // futur arc reserve
959  const int nbTasksThreads = 40;
960  const auto sizeBackBone = abs(stop - begin);
961  const auto chunkSize = getChunkSize(sizeBackBone, nbTasksThreads);
962  const auto chunkNb = getChunkCount(sizeBackBone, nbTasksThreads);
963  // si pas efficace vecteur de la taille de node ici a la place de acc
964  SimplexId tot = 0;
965  for(SimplexId chunkId = 0; chunkId < chunkNb; ++chunkId) {
966 #ifdef TTK_ENABLE_OPENMP
967 #pragma omp task firstprivate(chunkId) shared(trunkVerts, tot) \
968  OPTIONAL_PRIORITY(isPrior())
969 #endif
970  {
971  idNode lastVertInRange = 0;
972  SimplexId acc = 0;
973 
974  const SimplexId lowerBound = begin + chunkId * chunkSize;
975  const SimplexId upperBound
976  = min(stop, (begin + (chunkId + 1) * chunkSize));
977  for(SimplexId v = lowerBound; v < upperBound; ++v) {
978  const SimplexId s
979  = isST() ? scalars_->sortedVertices[lowerBound + upperBound - 1 - v]

985  updateCorrespondingArc(s, thisArc);
986 
987  if(params_->segm) {
988  if(oldVertInRange == lastVertInRange) {
989  ++acc;
990  } else {
991  // accumulated to have only one atomic update when needed
992  const idSuperArc oldArc
993  = upArcFromVert(trunkVerts[oldVertInRange]);
994  getSuperArc(oldArc)->atomicIncVisited(acc);
995 #ifdef TTK_ENABLE_FTM_TREE_PROCESS_SPEED
996 #ifdef TTK_ENABLE_OPENMP
997 #pragma omp atomic update
998 #endif
999  tot += acc;
1000 #endif
1001  acc = 1;
1002  }
1003  }
1004  }
1005  }
1006  // force increment last arc
1007  const idNode baseNode
1008  = getCorrespondingNodeId(trunkVerts[lastVertInRange]);
1009  const idSuperArc upArc = getNode(baseNode)->getUpSuperArcId(0);
1010  getSuperArc(upArc)->atomicIncVisited(acc);
1011 #ifdef TTK_ENABLE_FTM_TREE_PROCESS_SPEED
1012 #ifdef TTK_ENABLE_OPENMP
1013 #pragma omp atomic update
1014 #endif
1015  tot += acc;
1016 #endif

1020 #pragma omp taskwait
