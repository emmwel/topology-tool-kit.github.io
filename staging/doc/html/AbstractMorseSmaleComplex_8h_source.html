489 template <typename triangulationType>
490 int ttk::AbstractMorseSmaleComplex::getDescendingSeparatrices1(
491  const std::vector<Cell> &criticalPoints,
492  std::vector<Separatrix> &separatrices,
493  std::vector<std::vector<Cell>> &separatricesGeometry,
494  const triangulationType &triangulation) const {
495 
496  std::vector<SimplexId> saddleIndexes;
497  const SimplexId numberOfCriticalPoints = criticalPoints.size();
498  for(SimplexId i = 0; i < numberOfCriticalPoints; ++i) {
499  const Cell &criticalPoint = criticalPoints[i];
500 
501  if(criticalPoint.dim_ == 1)
502  saddleIndexes.push_back(i);
503  }
504  const SimplexId numberOfSaddles = saddleIndexes.size();
505 
506  // estimation of the number of separatrices, apriori :
507  // numberOfAscendingPaths=2, numberOfDescendingPaths=2
508  const SimplexId numberOfSeparatrices = 4 * numberOfSaddles;
509  separatrices.resize(numberOfSeparatrices);
510  separatricesGeometry.resize(numberOfSeparatrices);
511 
512  // apriori: by default construction, the separatrices are not valid
513 #ifdef TTK_ENABLE_OPENMP
514 #pragma omp parallel for num_threads(threadNumber_)
515 #endif
516  for(SimplexId i = 0; i < numberOfSaddles; ++i) {
517  const SimplexId saddleIndex = saddleIndexes[i];
518  const Cell &saddle = criticalPoints[saddleIndex];
519 
520  // add descending vpaths
521  {
522  const Cell &saddle1 = saddle;
523 
524  for(int j = 0; j < 2; ++j) {
525  const int shift = j + 2;
526 
527  SimplexId vertexId;
528  triangulation.getEdgeVertex(saddle1.id_, j, vertexId);
529 
530  std::vector<Cell> vpath;
531  vpath.push_back(saddle1);
532  discreteGradient_.getDescendingPath(
533  Cell(0, vertexId), vpath, triangulation);
534 
535  const Cell &lastCell = vpath.back();
536  if(lastCell.dim_ == 0 and discreteGradient_.isCellCritical(lastCell)) {
537  const SimplexId separatrixIndex = 4 * i + shift;
538 
539  separatricesGeometry[separatrixIndex] = std::move(vpath);
540  separatrices[separatrixIndex]
541  = Separatrix(true, saddle, lastCell, false, separatrixIndex);
542  }
543  }
544  }
545  }
546 
547  return 0;
548 }

750 template <typename triangulationType>
751 int ttk::AbstractMorseSmaleComplex::setAscendingSegmentation(
752  const std::vector<Cell> &criticalPoints,
753  std::vector<SimplexId> &maxSeeds,
754  SimplexId *const morseSmaleManifold,
755  SimplexId &numberOfMaxima,
756  const triangulationType &triangulation) const {
757 
758  const SimplexId numberOfVertices = triangulation.getNumberOfVertices();
759  std::fill(morseSmaleManifold, morseSmaleManifold + numberOfVertices, -1);
760 
761  const SimplexId numberOfCells = triangulation.getNumberOfCells();
762  std::vector<SimplexId> morseSmaleManifoldOnCells(numberOfCells, -1);
763  const int cellDim = triangulation.getDimensionality();
764 
765  // get the seeds : maxima
766  const SimplexId numberOfCriticalPoints = criticalPoints.size();
767  for(SimplexId i = 0; i < numberOfCriticalPoints; ++i) {
768  const Cell &criticalPoint = criticalPoints[i];
769 
770  if(criticalPoint.dim_ == cellDim)
771  maxSeeds.push_back(criticalPoint.id_);
772  }
773  const SimplexId numberOfSeeds = maxSeeds.size();
774  numberOfMaxima = numberOfSeeds;
775 
776  // Triangulation method pointers for 3D
777  auto getCellFace = &triangulationType::getCellTriangle;
778  auto getFaceStarNumber = &triangulationType::getTriangleStarNumber;
779  auto getFaceStar = &triangulationType::getTriangleStar;
780  if(cellDim == 2) {
781  // Triangulation method pointers for 2D
782  getCellFace = &triangulationType::getCellEdge;
783  getFaceStarNumber = &triangulationType::getEdgeStarNumber;
784  getFaceStar = &triangulationType::getEdgeStar;
785  }
786 
787 #ifdef TTK_ENABLE_OPENMP
788 #pragma omp parallel for num_threads(threadNumber_) schedule(dynamic)
789 #endif
790  for(SimplexId i = 0; i < numberOfSeeds; ++i) {
791  std::queue<SimplexId> bfs;
792 
793  // push the seed
794  {
795  const SimplexId seedId = maxSeeds[i];
796  bfs.push(seedId);
797  }
798 
799  // BFS traversal
800  while(!bfs.empty()) {
801  const SimplexId cofacetId = bfs.front();
802  bfs.pop();
803 
804  if(morseSmaleManifoldOnCells[cofacetId] == -1) {
805  morseSmaleManifoldOnCells[cofacetId] = i;
806 
807  for(int j = 0; j < (cellDim + 1); ++j) {
808  SimplexId facetId = -1;
809  (triangulation.*getCellFace)(cofacetId, j, facetId);
810 
811  SimplexId starNumber = (triangulation.*getFaceStarNumber)(facetId);
812  for(SimplexId k = 0; k < starNumber; ++k) {
813  SimplexId neighborId = -1;
814  (triangulation.*getFaceStar)(facetId, k, neighborId);
815 
816  if(neighborId == cofacetId) {
817  continue;
818  }
819 
820  const SimplexId pairedCellId = discreteGradient_.getPairedCell(
821  Cell(cellDim, neighborId), triangulation, true);
822 
823  if(pairedCellId == facetId)
824  bfs.push(neighborId);
825  }
826  }
827  }
828  }
829  }
830 
831  // put segmentation infos from cells to points
832 #ifdef TTK_ENABLE_OPENMP
833 #pragma omp parallel for num_threads(threadNumber_)
834 #endif
835  for(SimplexId i = 0; i < numberOfVertices; ++i) {
836  SimplexId starId;
837  triangulation.getVertexStar(i, 0, starId);
838  morseSmaleManifold[i] = morseSmaleManifoldOnCells[starId];
839  }
840 
841  return 0;
842 }

870  std::queue<SimplexId> bfs;
871 
872  // push the seed
873  {
874  const SimplexId seedId = seeds[i];
875  bfs.push(seedId);
876  }
877

901  Cell(0, neighborId), triangulation);
902

904  bfs.push(neighborId);
905  }
906  }
907  }
908  }
909  }

914 template <typename triangulationType>
915 int ttk::AbstractMorseSmaleComplex::setFinalSegmentation(
916  const SimplexId numberOfMaxima,
917  const SimplexId numberOfMinima,
918  const SimplexId *const ascendingManifold,
919  const SimplexId *const descendingManifold,
920  SimplexId *const morseSmaleManifold,
921  const triangulationType &triangulation) const {
922 
923  const size_t nVerts = triangulation.getNumberOfVertices();
924 
925  // associate a unique "sparse region id" to each (ascending, descending) pair
926 
927 #ifdef TTK_ENABLE_OPENMP
928 #pragma omp parallel for num_threads(threadNumber_)
929 #endif // TTK_ENABLE_OPENMP
930  for(size_t i = 0; i < nVerts; ++i) {
931  const auto d = ascendingManifold[i];
932  const auto a = descendingManifold[i];
933  if(a == -1 || d == -1) {
934  morseSmaleManifold[i] = -1;
935  } else {
936  morseSmaleManifold[i] = a * numberOfMaxima + d;
937  }
938  }
939 
940  // store the "sparse region ids" by copying the morseSmaleManifold output
941  std::vector<SimplexId> sparseRegionIds(
942  morseSmaleManifold, morseSmaleManifold + nVerts);
943 
944  // get unique "sparse region ids"
945  PSORT(sparseRegionIds.begin(), sparseRegionIds.end());
946  const auto last = std::unique(sparseRegionIds.begin(), sparseRegionIds.end());
947  sparseRegionIds.erase(last, sparseRegionIds.end());
948 
949  // "sparse region id" -> "dense region id"
950  std::map<SimplexId, size_t> sparseToDenseRegionId{};
951 
952  for(size_t i = 0; i < sparseRegionIds.size(); ++i) {
953  sparseToDenseRegionId[sparseRegionIds[i]] = i;
954  }
955 
956  // update region id on all vertices: "sparse id" -> "dense id"
957 
958 #ifdef TTK_ENABLE_OPENMP
959 #pragma omp parallel for num_threads(threadNumber_)
960 #endif // TTK_ENABLE_OPENMP
961  for(size_t i = 0; i < nVerts; ++i) {
962  morseSmaleManifold[i] = sparseToDenseRegionId[morseSmaleManifold[i]];
963  }
964 
965  return 0;
966 }
