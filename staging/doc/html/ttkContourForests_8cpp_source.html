701 int ttkContourForests::sample(unsigned int samplingLevel) {
702  samples_.resize(3);
703  samples_[static_cast<int>(treeType_)].resize(tree_->getNumberOfSuperArcs());
704  vector<vector<SimplexId>> sampleList(samplingLevel);
705 
706  SuperArc *a;
707  for(SimplexId i = 0; i < (SimplexId)tree_->getNumberOfSuperArcs(); ++i) {
708  a = tree_->getSuperArc(i);
709 
710  if(!a->isPruned()) {
711  for(unsigned int j = 0; j < samplingLevel; ++j) {
712  sampleList[j].clear();
713  }
714 
715  double fmax, fmin;
716  SimplexId nodeMaxId, nodeMinId;
717  SimplexId nodeMaxVId, nodeMinVId;
718  double delta;
719  if(a->getNumberOfRegularNodes()) {
720  if(treeType_ == TreeType::Split) {
721  nodeMaxId = a->getDownNodeId();
722  nodeMinId = a->getUpNodeId();
723  } else {
724  nodeMaxId = a->getUpNodeId();
725  nodeMinId = a->getDownNodeId();
726  }
727 
728  nodeMaxVId = tree_->getNode(nodeMaxId)->getVertexId();
729  nodeMinVId = tree_->getNode(nodeMinId)->getVertexId();
730 
731  fmax = vertexScalars_[nodeMaxVId];
732  fmin = vertexScalars_[nodeMinVId];
733 
734  delta = (fmax - fmin) / samplingLevel;
735 
736  double f;
737  SimplexId nodeId;
738  SimplexId vertexId;
739  for(SimplexId j = 0; j < a->getNumberOfRegularNodes(); ++j) {
740  nodeId = a->getRegularNodeId(j);
741  if(a->isMasqued(j))
742  continue;
743  vertexId = nodeId;
744  f = vertexScalars_[vertexId];
745 
746  for(unsigned int k = 0; k < samplingLevel; ++k) {
747  if(f <= (k + 1) * delta + fmin) {
748  sampleList[k].push_back(nodeId);
749  break;
750  }
751  }
752  }
753 
754  // update the arc
755  for(SimplexId j = 0; j < (SimplexId)sampleList.size(); ++j)
756  samples_[static_cast<int>(treeType_)][i].push_back(sampleList[j]);
757  }
758  }
759  }

831  vector<vector<double>> barycenterList(N);

833  barycenterList[i].resize(3);

835  SimplexId up_vId;
836  SimplexId down_vId;

839  ->getVertexId();

867  + barycenters_[static_cast<int>(treeType_)][idArc][i + 1][k])
868  * 0.5;

871  for(unsigned int k = 0; k < 3; ++k)
872  barycenterList[N - 1][k]
873  = (p1[k] + barycenters_[static_cast<int>(treeType_)][idArc][N - 1][k])
874  * 0.5;
875  } else {
876  for(unsigned int k = 0; k < 3; ++k)
877  barycenterList[0][k] = (p0[k] + p1[k]) * 0.5;
878  }
879 
880  // copy //
881  for(int i = 0; i < N; ++i) {
882  for(unsigned int k = 0; k < 3; ++k)
883  barycenters_[static_cast<int>(treeType_)][idArc][i][k]
